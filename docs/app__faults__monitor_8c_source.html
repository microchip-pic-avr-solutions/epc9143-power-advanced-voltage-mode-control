82 volatile uint16_t appFaultMonitor_Execute(void)
83 {
84  volatile uint16_t retval=1;
85  static bool pre_fault_active=false;
86 
87  // Call fault handler
88  retval &= drv_FaultCheck(&fltobj_BuckUVLO);
89  retval &= drv_FaultCheck(&fltobj_BuckOVLO);
90  retval &= drv_FaultCheck(&fltobj_BuckRegErr);
91  retval &= drv_FaultCheck(&fltobj_BuckOCP);
92 
93  // Combine individual fault bits to a common fault indicator
94  buck.status.bits.fault_active = (bool) (
95  fltobj_BuckUVLO.Status.bits.FaultStatus |
96  fltobj_BuckOVLO.Status.bits.FaultStatus |
97  fltobj_BuckRegErr.Status.bits.FaultStatus |
98  fltobj_BuckOCP.Status.bits.FaultStatus
99  );
100 
101  // If system has recovered from a global fault condition,
102  // trigger fault recovery function
103  if ((pre_fault_active) && (!buck.status.bits.fault_active))
104  { retval = appPowerSupply_Resume(); }
105 
106  // Track global fault bit transitions
107  pre_fault_active = buck.status.bits.fault_active;
108 
109 
110  return (retval);
111 }

162 volatile uint16_t uvlo_FaultInitialize(void)
163 {
164  volatile uint16_t retval=1;
165 
166  // Initialize UVLO fault object
167  fltobj_BuckUVLO = fltobjClear; // Pre-initialize fault object
168 
169  fltobj_BuckUVLO.SourceObject.ptrObject = &buck.data.v_in; // Set pointer to variable to monitor
170  fltobj_BuckUVLO.SourceObject.bitMask = 0xFFFF; // Compare all bits of SOURCE (no bit filter)
171  fltobj_BuckUVLO.ReferenceObject.ptrObject = NULL; // Clear pointer to "compare against" variable
172  fltobj_BuckUVLO.ReferenceObject.bitMask = 0xFFFF; // Clear pointer to "compare against" variable
173 
174  fltobj_BuckUVLO.Status.bits.CompareType = FLTCMP_LESS_THAN; // Select Compare-Type
175 
176  fltobj_BuckUVLO.TripResponse.compareThreshold = BUCK_VIN_UVLO_TRIP; // Set fault trip level
177  fltobj_BuckUVLO.TripResponse.eventThreshold = BUCK_UVLO_TDLY; // Set counter level at which a FAULT condition will be tripped
178  fltobj_BuckUVLO.TripResponse.ptrResponseFunction = &appPowerSupply_Suspend; // Set pointer to user-function which should be called when a FAULT is tripped
179 
180  fltobj_BuckUVLO.RecoveryResponse.compareThreshold = BUCK_VIN_UVLO_RELEASE; // Set fault recovery level
181  fltobj_BuckUVLO.RecoveryResponse.eventThreshold = BUCK_UVLO_RDLY; // Set counter level at which a FAULT condition will be cleared
182  fltobj_BuckUVLO.RecoveryResponse.ptrResponseFunction = NULL; // Clear recovery function pointer
183 
184  fltobj_BuckUVLO.Counter = 0; // Clear fault event counter
185  fltobj_BuckUVLO.Status.bits.FaultActive = true; // Set fault condition flag (must be cleared by fault check)
186  fltobj_BuckUVLO.Status.bits.FaultStatus = true; // Set fault flag (must be cleared by fault check)
187  fltobj_BuckUVLO.Status.bits.Enabled = true; // Enable fault checks
188 
189  return(retval);
190 
191 }
