78 volatile uint16_t appFaultMonitor_Execute(void)
79 {
80  volatile uint16_t retval=1;
81  static bool pre_fault_active=false;
82 
83  // Call fault handler
84  retval &= drv_FaultCheck(&fltobj_BuckUVLO);
85  retval &= drv_FaultCheck(&fltobj_BuckOVLO);
86  retval &= drv_FaultCheck(&fltobj_BuckRegErr);
87  retval &= drv_FaultCheck(&fltobj_BuckOCP);
88 
89  // Combine individual fault bits to a common fault indicator
90  buck.status.bits.fault_active = (bool) (
91  fltobj_BuckUVLO.Status.bits.FaultStatus |
92  fltobj_BuckOVLO.Status.bits.FaultStatus |
93  fltobj_BuckRegErr.Status.bits.FaultStatus |
94  fltobj_BuckOCP.Status.bits.FaultStatus
95  );
96 
97  // If system has recovered from a global fault condition,
98  // trigger fault recovery function
99  if ((pre_fault_active) && (!buck.status.bits.fault_active))
100  { retval = appPowerSupply_Resume(); }
101 
102  // Track global fault bit transitions
103  pre_fault_active = buck.status.bits.fault_active;
104 
105 
106  return (retval);
107 }

154 volatile uint16_t uvlo_FaultInitialize(void)
155 {
156  volatile uint16_t retval=1;
157 
158  // Initialize UVLO fault object
159  fltobj_BuckUVLO = fltobjClear; // Pre-initialize fault object
160 
161  fltobj_BuckUVLO.SourceObject.ptrObject = &buck.data.v_in; // Set pointer to variable to monitor
162  fltobj_BuckUVLO.SourceObject.bitMask = 0xFFFF; // Compare all bits of SOURCE (no bit filter)
163  fltobj_BuckUVLO.ReferenceObject.ptrObject = NULL; // Clear pointer to "compare against" variable
164  fltobj_BuckUVLO.ReferenceObject.bitMask = 0xFFFF; // Clear pointer to "compare against" variable
165 
166  fltobj_BuckUVLO.Status.bits.CompareType = FLTCMP_LESS_THAN; // Select Compare-Type
167 
168  fltobj_BuckUVLO.TripResponse.compareThreshold = BUCK_VIN_UVLO_TRIP; // Set fault trip level
169  fltobj_BuckUVLO.TripResponse.eventThreshold = BUCK_UVLO_TDLY; // Set counter level at which a FAULT condition will be tripped
170  fltobj_BuckUVLO.TripResponse.ptrResponseFunction = &appPowerSupply_Suspend; // Set pointer to user-function which should be called when a FAULT is tripped
171 
172  fltobj_BuckUVLO.RecoveryResponse.compareThreshold = BUCK_VIN_UVLO_RELEASE; // Set fault recovery level
173  fltobj_BuckUVLO.RecoveryResponse.eventThreshold = BUCK_UVLO_RDLY; // Set counter level at which a FAULT condition will be cleared
174  fltobj_BuckUVLO.RecoveryResponse.ptrResponseFunction = NULL; // Clear recovery function pointer
175 
176  fltobj_BuckUVLO.Counter = 0; // Clear fault event counter
177  fltobj_BuckUVLO.Status.bits.FaultActive = true; // Set fault condition flag (must be cleared by fault check)
178  fltobj_BuckUVLO.Status.bits.FaultStatus = true; // Set fault flag (must be cleared by fault check)
179  fltobj_BuckUVLO.Status.bits.Enabled = true; // Enable fault checks
180 
181  return(retval);
182 
183 }
