71 volatile uint16_t State_Initialize(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
72 {
73  volatile uint16_t _i=0;
74 
75  buckInstance->startup.power_on_delay.counter = 0; // Reset power on counter
76  buckInstance->startup.power_good_delay.counter = 0; // Reset power good counter
77 
78  // Reset all status bits
79  buckInstance->status.bits.adc_active = false;
80 
81  // Initiate current sensor calibration flag bit
82  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_VMC)
83  buckInstance->status.bits.cs_calib_complete = true;
84  else if (buckInstance->status.bits.cs_calib_enable)
85  buckInstance->status.bits.cs_calib_complete = false;
86  else
87  buckInstance->status.bits.cs_calib_complete = true;
88 
89  // Disable control loops
90  buckInstance->v_loop.controller->status.bits.enabled = false; // Disable voltage loop
91 
92  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_ACMC) // In current mode...
93  {
94  for (_i=0; _i<buckInstance->set_values.no_of_phases; _i++) {
95  buckInstance->i_loop[_i].controller->status.bits.enabled = false; // Disable current loop
96  }
97  }
98 
99  // Clear busy bit
100  buckInstance->status.bits.busy = false; // Clear BUSY bit
101  buckInstance->status.bits.ready = true; // Set READY bit indicating state machine has passed INITIALIZED state
102 
103  // Transition to STATE_RESET
104  return(BUCK_OPSRET_COMPLETE);
105 
106 }

120 volatile uint16_t State_Reset(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
121 {
122  volatile uint16_t retval=1;
123  volatile uint16_t _i=0;
124 
125  // Disable all PWM outputs & control loops (immediate power cut-off)
126  retval &= buckPWM_Suspend(buckInstance); // Disable PWM outputs
127 
128  // Disable voltage loop controller and reset control loop histories
129  buckInstance->v_loop.controller->status.bits.enabled = false; // disable voltage control loop
130  buckInstance->v_loop.ctrl_Reset(buckInstance->v_loop.controller); // Reset control histories of outer voltage controller
131  *buckInstance->v_loop.controller->Ports.Target.ptrAddress =
132  buckInstance->v_loop.controller->Limits.MinOutput;
133 
134  // Disable current loop controller and reset control loop histories
135  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_ACMC)
136  { // Disable all current control loops and reset control loop histories
137  for (_i=0; _i<buckInstance->set_values.no_of_phases; _i++) {
138  buckInstance->i_loop[_i].controller->status.bits.enabled = false;
139  buckInstance->i_loop[_i].ctrl_Reset(buckInstance->i_loop[_i].controller);
140  *buckInstance->i_loop[_i].controller->Ports.Target.ptrAddress =
141  buckInstance->i_loop[_i].controller->Limits.MinOutput;
142  }
143  }
144 
145  // Reset the bulk voltage settling counters
146  buckInstance->startup.power_on_delay.counter = 0; // Clear Power On Delay counter
147  buckInstance->startup.power_good_delay.counter = 0; // Clear Power Good Delay counter
148 
149  // Reset all status bits
150  buckInstance->status.bits.adc_active = false;
151  buckInstance->status.bits.busy = false; // Clear BUSY bit
152 
153  // If any sub-function call went unsuccessful, reset state machine
154  // else, move on to next state
155 
156  if (retval)
157  return(BUCK_OPSRET_COMPLETE);
158  else
159  return(BUCK_OPSRET_ERROR);
160 
161 }

175 volatile uint16_t State_Standby(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
176 {
177  // if the 'autorun' option is set, automatically set the GO bit when the
178  // converter is enabled
179  if ((buckInstance->status.bits.enabled) && (buckInstance->status.bits.autorun))
180  { buckInstance->status.bits.GO = true; }
181 
182  // Wait for all startup conditions to be met
183  if ((buckInstance->status.bits.enabled) && // state machine needs to be enabled
184  (buckInstance->status.bits.GO) && // GO-bit needs to be set
185  (buckInstance->status.bits.adc_active) && // ADC needs to be running
186  (buckInstance->status.bits.pwm_active) && // PWM needs to be running
187  (!buckInstance->status.bits.fault_active) && // No active fault is present
188  (!buckInstance->status.bits.suspend) && // Power supply is not held in suspend mode
189  (buckInstance->status.bits.cs_calib_complete) // Current Sensor Calibration complete
190  )
191  {
192  buckInstance->status.bits.GO = false;
193  return(BUCK_OPSRET_COMPLETE);
194  }
195  else
196  // Remain in current state until bit-test becomes true
197  {
198  return(BUCK_OPSRET_REPEAT);
199  }
200 
201 }

217 volatile uint16_t State_RampUp(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
218 {
219  volatile uint16_t retval=0;
220 
221  // If sub-state pointer index is out of range, reset to ZERO
222  if (buckInstance->state_id.bits.substate_id >= BuckRampUpSubStateList_size)
223  buckInstance->state_id.bits.substate_id = 0;
224 
225  // If selected sub-state index contains a NULL-pointer, exit here
226  if (BuckConverterRampUpSubStateMachine[buckInstance->state_id.bits.substate_id] == NULL)
227  return(BUCK_OPSRET_ERROR);
228 
229  // Execute ramp-up sub-state
230  retval = BuckConverterRampUpSubStateMachine[buckInstance->state_id.bits.substate_id](buckInstance);
231 
232  // Validate sub-state function return
233  switch (retval)
234  {
235  // If the sub-state returns a REPEAT, leave the sub-state pointer at the recent
236  // index and return REPEAT to the outer state machine to call State_RampUp again
237  case BUCK_OPSRET_REPEAT:
238 
239  retval = BUCK_OPSRET_REPEAT;
240 
241  break;
242 
243  // If the sub-state returns a COMPLETE, increment the sub-state pointer to call the
244  // next sub-state next time and return REPEAT to the outer state machine to call
245  // State_RampUp again
246  case BUCK_OPSRET_COMPLETE:
247 
248  // Increment sub-state pointer by one tick
249  buckInstance->state_id.bits.substate_id++;
250 
251  // CHeck if pointer is out of range
252  if (buckInstance->state_id.bits.substate_id < BuckRampUpSubStateList_size)
253  { // if execution list is not complete yet, return op-state as REPEAT
254  retval = BUCK_OPSRET_REPEAT;
255 
256  if (buckInstance->state_id.bits.substate_id == BUCK_OPSTATE_I_RAMP_UP)
257  { // if the next state is set to "I Ramp-Up", check if converter supports
258  // this sub-state in its current configuration and step over it if not
259 
260  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_VMC)
261  buckInstance->state_id.bits.substate_id++; // Increment sub-state pointer
262 
263  }
264 
265  }
266  else
267  {
268  // if last item of execution list is complete. return op-state as COMLETE
269  retval = BUCK_OPSRET_COMPLETE;
270  }
271 
272  break;
273 
274  // If the sub-state returns any other value, something went wrong. In this case
275  // return ERROR to outer state machine triggering a main state machine reset.
276  default:
277  // if any other return value is received, switch immediately to ERROR
278  retval = BUCK_OPSRET_ERROR;
279 
280  break;
281  }
282 
283  return(retval);
284 }

307 volatile uint16_t State_Online(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
308 {
309  if(buckInstance->set_values.v_ref != buckInstance->v_loop.reference)
310  {
311  // Set the BUSY bit indicating a delay/ramp period being executed
312  buckInstance->status.bits.busy = true;
313 
314  // New reference value is less than controller reference value => ramp down
315  if(buckInstance->set_values.v_ref < buckInstance->v_loop.reference){
316  // decrement reference until new reference level is reached
317  buckInstance->v_loop.reference -= buckInstance->startup.v_ramp.ref_inc_step; // decrement reference
318  if(buckInstance->v_loop.reference < buckInstance->set_values.v_ref) { // check if ramp is complete
319  buckInstance->v_loop.reference = buckInstance->set_values.v_ref; // clamp reference level to setting
320  }
321 
322  }
323  // New reference value is greater than controller reference value => ramp up
324  else if(buckInstance->set_values.v_ref > buckInstance->v_loop.reference){
325  // increment reference until new reference level is reached
326  buckInstance->v_loop.reference += buckInstance->startup.v_ramp.ref_inc_step; // increment reference
327  if(buckInstance->v_loop.reference > buckInstance->set_values.v_ref) { // check if ramp is complete
328  buckInstance->v_loop.reference = buckInstance->set_values.v_ref; // clamp reference level to setting
329  }
330 
331  }
332 
333  }
334  else{
335  // Clear the BUSY bit indicating "no state machine activity"
336  buckInstance->status.bits.busy = false;
337  }
338 
339  // remain in STATE_ONLINE
340  return(BUCK_OPSRET_REPEAT);
341 
342 }

355 volatile uint16_t State_Error(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
356 {
357  volatile uint16_t retval=0;
358 
359  // If this function is ever called, bring state machine back on track
360  // by resetting it to INITIALIZE
361 
362  retval = buckPWM_Suspend(buckInstance); // Hold PWM output (turning off power)
363  buckInstance->status.bits.busy = false; // Reset busy bit
364 
365  if(retval)
366  retval = BUCK_OPSRET_COMPLETE;
367  else
368  retval = BUCK_OPSRET_ERROR;
369 
370 
371  return(retval);
372 }
