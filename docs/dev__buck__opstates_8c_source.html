66 volatile uint16_t State_Initialize(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
67 {
68  volatile uint16_t _i=0;
69 
70  buckInstance->startup.power_on_delay.counter = 0; // Reset power on counter
71  buckInstance->startup.power_good_delay.counter = 0; // Reset power good counter
72 
73  // Reset all status bits
74  buckInstance->status.bits.adc_active = false;
75 
76  // Initiate current sensor calibration flag bit
77  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_VMC)
78  buckInstance->status.bits.cs_calib_complete = true;
79  else if (buckInstance->status.bits.cs_calib_enable)
80  buckInstance->status.bits.cs_calib_complete = false;
81  else
82  buckInstance->status.bits.cs_calib_complete = true;
83 
84  // Disable control loops
85  buckInstance->v_loop.controller->status.bits.enabled = false; // Disable voltage loop
86 
87  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_ACMC) // In current mode...
88  {
89  for (_i=0; _i<buckInstance->set_values.no_of_phases; _i++) {
90  buckInstance->i_loop[_i].controller->status.bits.enabled = false; // Disable current loop
91  }
92  }
93 
94  // Clear busy bit
95  buckInstance->status.bits.busy = false; // Clear BUSY bit
96  buckInstance->status.bits.ready = true; // Set READY bit indicating state machine has passed INITIALIZED state
97 
98  // Transition to STATE_RESET
99  return(BUCK_OPSRET_COMPLETE);
100 
101 }

117 volatile uint16_t State_Reset(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
118 {
119  volatile uint16_t retval=1;
120  volatile uint16_t _i=0;
121 
122  // Disable all PWM outputs & control loops (immediate power cut-off)
123  retval &= buckPWM_Suspend(buckInstance); // Disable PWM outputs
124 
125  // Disable voltage loop controller and reset control loop histories
126  buckInstance->v_loop.controller->status.bits.enabled = false; // disable voltage control loop
127  buckInstance->v_loop.ctrl_Reset(buckInstance->v_loop.controller); // Reset control histories of outer voltage controller
128  *buckInstance->v_loop.controller->Ports.Target.ptrAddress =
129  buckInstance->v_loop.controller->Limits.MinOutput;
130 
131  // Disable current loop controller and reset control loop histories
132  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_ACMC)
133  { // Disable all current control loops and reset control loop histories
134  for (_i=0; _i<buckInstance->set_values.no_of_phases; _i++) {
135  buckInstance->i_loop[_i].controller->status.bits.enabled = false;
136  buckInstance->i_loop[_i].ctrl_Reset(buckInstance->i_loop[_i].controller);
137  *buckInstance->i_loop[_i].controller->Ports.Target.ptrAddress =
138  buckInstance->i_loop[_i].controller->Limits.MinOutput;
139  }
140  }
141 
142  // Reset the bulk voltage settling counters
143  buckInstance->startup.power_on_delay.counter = 0; // Clear Power On Delay counter
144  buckInstance->startup.power_good_delay.counter = 0; // Clear Power Good Delay counter
145 
146  // Reset all status bits
147  buckInstance->status.bits.adc_active = false;
148  buckInstance->status.bits.busy = false; // Clear BUSY bit
149 
150  // If any sub-function call went unsuccessful, reset state machine
151  // else, move on to next state
152 
153  if (retval)
154  return(BUCK_OPSRET_COMPLETE);
155  else
156  return(BUCK_OPSRET_ERROR);
157 
158 }

173 volatile uint16_t State_Standby(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
174 {
175  volatile uint16_t retval=0;
176 
177  // if the 'autorun' option is set, automatically set the GO bit when the
178  // converter is enabled
179  if ((buckInstance->status.bits.enabled) && (buckInstance->status.bits.autorun))
180  { buckInstance->status.bits.GO = true; }
181 
182  // If current sense feedback offset calibration is enabled,
183  // call calibration sub-state routine
184  retval = drv_BuckConverter_SpecialFunctionExecute(buckInstance, CS_OFSET_CALIBRATION);
185 
186  switch (retval)
187  {
188  case BUCK_OPSRET_COMPLETE:
189  buckInstance->status.bits.cs_calib_complete = true; // Set CS Calibration Flag to COMPLETE
190  buckInstance->status.bits.busy = false; // Clear BUSY bit indicating on-going activity
191  break;
192  case BUCK_OPSRET_REPEAT:
193  buckInstance->status.bits.cs_calib_complete = false; // Set CS Calibration Flag to COMPLETE
194  buckInstance->status.bits.busy = true; // Set BUSY bit indicating on-going activity
195  return(BUCK_OPSRET_REPEAT);
196  break;
197  default:
198  buckInstance->status.bits.cs_calib_complete = false; // Set CS Calibration Flag to COMPLETE
199  buckInstance->status.bits.busy = false; // Clear BUSY bit indicating on-going activity
200  return(BUCK_OPSRET_ERROR);
201  break;
202  }
203 
204  // Wait for all startup conditions to be met
205  if ((buckInstance->status.bits.enabled) && // state machine needs to be enabled
206  (buckInstance->status.bits.GO) && // GO-bit needs to be set
207  (buckInstance->status.bits.adc_active) && // ADC needs to be running
208  (buckInstance->status.bits.pwm_active) && // PWM needs to be running
209  (!buckInstance->status.bits.fault_active) && // No active fault is present
210  (!buckInstance->status.bits.suspend) && // Power supply is not held in suspend mode
211  (buckInstance->status.bits.cs_calib_complete) // Current Sensor Calibration complete
212  )
213  {
214  buckInstance->status.bits.GO = false;
215  return(BUCK_OPSRET_COMPLETE);
216  }
217  else
218  // Remain in current state until bit-test becomes true
219  {
220  return(BUCK_OPSRET_REPEAT);
221  }
222 
223 }

239 volatile uint16_t State_RampUp(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
240 {
241  volatile uint16_t retval=0;
242 
243  // If sub-state pointer index is out of range, reset to ZERO
244  if (buckInstance->state_id.bits.substate_id >= BuckRampUpSubStateList_size)
245  buckInstance->state_id.bits.substate_id = 0;
246 
247  // If selected sub-state index contains a NULL-pointer, exit here
248  if (BuckConverterRampUpSubStateMachine[buckInstance->state_id.bits.substate_id] == NULL)
249  return(BUCK_OPSRET_ERROR);
250 
251  // Execute ramp-up sub-state
252  retval = BuckConverterRampUpSubStateMachine[buckInstance->state_id.bits.substate_id](buckInstance);
253 
254  // Validate sub-state function return
255  switch (retval)
256  {
257  // If the sub-state returns a REPEAT, leave the sub-state pointer at the recent
258  // index and return REPEAT to the outer state machine to call State_RampUp again
259  case BUCK_OPSRET_REPEAT:
260 
261  retval = BUCK_OPSRET_REPEAT;
262 
263  break;
264 
265  // If the sub-state returns a COMPLETE, increment the sub-state pointer to call the
266  // next sub-state next time and return REPEAT to the outer state machine to call
267  // State_RampUp again
268  case BUCK_OPSRET_COMPLETE:
269 
270  // Increment sub-state pointer by one tick
271  buckInstance->state_id.bits.substate_id++;
272 
273  // CHeck if pointer is out of range
274  if (buckInstance->state_id.bits.substate_id < BuckRampUpSubStateList_size)
275  { // if execution list is not complete yet, return op-state as REPEAT
276  retval = BUCK_OPSRET_REPEAT;
277 
278  if (buckInstance->state_id.bits.substate_id == BUCK_OPSTATE_I_RAMP_UP)
279  { // if the next state is set to "I Ramp-Up", check if converter supports
280  // this sub-state in its current configuration and step over it if not
281 
282  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_VMC)
283  buckInstance->state_id.bits.substate_id++; // Increment sub-state pointer
284 
285  }
286 
287  }
288  else
289  {
290  // if last item of execution list is complete. return op-state as COMPLETE
291  retval = BUCK_OPSRET_COMPLETE;
292  }
293 
294  break;
295 
296  // If the sub-state returns any other value, something went wrong. In this case
297  // return ERROR to outer state machine triggering a main state machine reset.
298  default:
299  // if any other return value is received, switch immediately to ERROR
300  retval = BUCK_OPSRET_ERROR;
301 
302  break;
303  }
304 
305  return(retval);
306 }

330 volatile uint16_t State_Online(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
331 {
332  if(buckInstance->set_values.v_ref != buckInstance->v_loop.reference)
333  {
334  // Set the BUSY bit indicating a delay/ramp period being executed
335  buckInstance->status.bits.busy = true;
336 
337  // New reference value is less than controller reference value => ramp down
338  if(buckInstance->set_values.v_ref < buckInstance->v_loop.reference){
339  // decrement reference until new reference level is reached
340  buckInstance->v_loop.reference -= buckInstance->startup.v_ramp.ref_inc_step; // decrement reference
341  if(buckInstance->v_loop.reference < buckInstance->set_values.v_ref) { // check if ramp is complete
342  buckInstance->v_loop.reference = buckInstance->set_values.v_ref; // clamp reference level to setting
343  }
344 
345  }
346  // New reference value is greater than controller reference value => ramp up
347  else if(buckInstance->set_values.v_ref > buckInstance->v_loop.reference){
348  // increment reference until new reference level is reached
349  buckInstance->v_loop.reference += buckInstance->startup.v_ramp.ref_inc_step; // increment reference
350  if(buckInstance->v_loop.reference > buckInstance->set_values.v_ref) { // check if ramp is complete
351  buckInstance->v_loop.reference = buckInstance->set_values.v_ref; // clamp reference level to setting
352  }
353 
354  }
355 
356  }
357  else{
358  // Clear the BUSY bit indicating "no state machine activity"
359  buckInstance->status.bits.busy = false;
360  }
361 
362  // remain in STATE_ONLINE
363  return(BUCK_OPSRET_REPEAT);
364 
365 }

379 volatile uint16_t State_Error(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
380 {
381  volatile uint16_t retval=0;
382 
383  // If this function is ever called, bring state machine back on track
384  // by resetting it to INITIALIZE
385 
386  retval = buckPWM_Suspend(buckInstance); // Hold PWM output (turning off power)
387  buckInstance->status.bits.busy = false; // Reset busy bit
388 
389  if(retval)
390  retval = BUCK_OPSRET_COMPLETE;
391  else
392  retval = BUCK_OPSRET_ERROR;
393 
394 
395  return(retval);
396 }
