73 volatile uint16_t SubState_PowerOnDelay(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
74 {
75  volatile uint16_t retval=0;
76 
77  // Set BUSY bit until process is complete
78  buckInstance->status.bits.busy = true;
79 
80  // delay startup until POWER ON DELAY has expired
81  if(buckInstance->startup.power_on_delay.counter++ > buckInstance->startup.power_on_delay.period)
82  {
83  // Clamp POD counter to EXPIRED
84  buckInstance->startup.power_on_delay.counter =
85  (buckInstance->startup.power_on_delay.period + 1); // Saturate power on counter
86 
87  retval = BUCK_OPSRET_COMPLETE;
88  }
89  else
90  // When the period has not expired yet, return to this function
91  {
92  retval = BUCK_OPSRET_REPEAT;
93  }
94 
95  return(retval);

117 volatile uint16_t SubState_PrepareVRampUp(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
118 {
119  volatile uint16_t _i=0;
120  volatile uint32_t _vout=0, _vin=0, _start_dc=0;
121 
122  // Set BUSY bit until process is complete
123  buckInstance->status.bits.busy = true;
124 
125  // Hijack voltage loop controller reference
126  buckInstance->startup.v_ramp.reference = 0; // Reset Soft-Start Voltage Reference
127  buckInstance->startup.i_ramp.reference = 0; // Reset Soft-Start Current Reference
128  buckInstance->v_loop.controller->Ports.ptrControlReference =
129  &buckInstance->startup.v_ramp.reference; // Voltage loop is pointing to Soft-Start Reference
130 
131  // Pre-charge reference
132  // Never start above the pre-biased output voltage.
133  // Always start at or slightly below the pre-biased output voltage
134  buckInstance->startup.v_ramp.reference = buckInstance->data.v_out;
135 
136  // In average current mode, set current reference limit to max startup current level
137  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_ACMC)
138  { // Disable all current control loops and reset control loop histories
139  buckInstance->v_loop.maximum = buckInstance->set_values.i_ref;
140  buckInstance->v_loop.controller->Limits.MaxOutput = buckInstance->v_loop.maximum;
141  }
142 
143  // Pre-charge PWM and control loop history
144  if(((buckInstance->data.v_in - buckInstance->feedback.ad_vin.scaling.offset) > 0) &&
145  ((buckInstance->data.v_out - buckInstance->feedback.ad_vout.scaling.offset) > 0) )
146  {
147  _vout = __builtin_muluu(
148  (buckInstance->data.v_out - buckInstance->feedback.ad_vout.scaling.offset),
149  buckInstance->feedback.ad_vout.scaling.factor);
150  _vout >>= (16 - buckInstance->feedback.ad_vout.scaling.scaler);
151 
152  _vin = __builtin_muluu(
153  (buckInstance->data.v_in - buckInstance->feedback.ad_vin.scaling.offset),
154  buckInstance->feedback.ad_vin.scaling.factor);
155  _vin >>= (16 - buckInstance->feedback.ad_vin.scaling.scaler);
156 
157  // Protect against negative duty cycle results
158  if (_vout < _vin) _vout = _vin;
159 
160  // CALCULATE BUCK CONVERTER STARTUP DUTY RATIO
161  // DC = (VOUT-VIN) / VOUT, where DC = D * PERIOD
162 
163  _start_dc = __builtin_muluu((_vout-_vin), buckInstance->sw_node[0].period);
164  _start_dc = __builtin_divud(_start_dc, (uint16_t)_vout);
165  }
166  else
167  // If there is no input voltage or no output voltage, start with minimum duty ratio
168  {
169  for (_i=0; _i<BUCK_NO_OF_PHASES; _i++) {
170  _start_dc = (uint16_t)buckInstance->sw_node[_i].duty_ratio_min;
171  }
172  }
173 
174  // Program PWM module and pre-charge control loop
175  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_VMC)
176  {
177  if(_start_dc < buckInstance->v_loop.minimum)
178  { _start_dc = buckInstance->v_loop.minimum; }
179  else if(_start_dc > buckInstance->v_loop.maximum)
180  { _start_dc = buckInstance->v_loop.maximum; }
181 
182  buckInstance->v_loop.ctrl_Precharge(buckInstance->v_loop.controller, 0, _start_dc);
183  *buckInstance->v_loop.controller->Ports.Target.ptrAddress = _start_dc; // set initial PWM duty ratio
184  if(buckInstance->v_loop.controller->Ports.AltTarget.ptrAddress != NULL)
185  *buckInstance->v_loop.controller->Ports.AltTarget.ptrAddress = _start_dc; // set initial PWM duty ratio
186 
187  }
188  else if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_ACMC)
189  {
190  // Limit startup duty cycle to largest minimum/smallest maximum
191  for (_i=0; _i<BUCK_NO_OF_PHASES; _i++) {
192 
193  if(_start_dc < buckInstance->i_loop[_i].minimum)
194  { _start_dc = buckInstance->i_loop[_i].minimum; }
195  else if(_start_dc > buckInstance->i_loop[_i].maximum)
196  { _start_dc = buckInstance->i_loop[_i].maximum; }
197  }
198 
199  // Set initial duty cycle
200  for (_i=0; _i<BUCK_NO_OF_PHASES; _i++)
201  {
202  // pre-charge current loop control histories with ideal duty cycle
203  buckInstance->i_loop[_i].ctrl_Precharge(
204  buckInstance->i_loop[_i].controller, 0, _start_dc
205  );
206 
207  // pre-charge PWM outputs with ideal duty cycle
208  *buckInstance->i_loop[_i].controller->Ports.Target.ptrAddress = _start_dc; // set initial PWM duty ratio
209  if(buckInstance->i_loop[_i].controller->Ports.AltTarget.ptrAddress != NULL)
210  *buckInstance->i_loop[_i].controller->Ports.AltTarget.ptrAddress = _start_dc; // set initial PWM duty ratio
211  }
212  }
213  else
214  {
215  return(BUCK_OPSRET_ERROR);
216  }
217 
218  return(BUCK_OPSRET_COMPLETE);
219 
220 }

238 volatile uint16_t SubState_VRampUp(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
239 {
240  volatile uint16_t retval=1;
241  volatile uint16_t _i=0;
242 
243  // Set BUSY bit until process is complete
244  buckInstance->status.bits.busy = true;
245 
246  // ensure control loop is enabled
247  if(!buckInstance->v_loop.controller->status.bits.enabled) {
248 
249  // Enable all PWM channels
250  retval &= buckPWM_Resume(buckInstance); // Enable PWM outputs
251 
252  // In voltage mode control...
253  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_VMC)
254  { // only enable voltage loop
255  buckInstance->v_loop.controller->status.bits.enabled = true; // enable voltage loop controller
256  }
257  // In average current mode control...
258  else if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_ACMC)
259  {
260  // enable voltage loop controller
261  buckInstance->v_loop.controller->status.bits.enabled = true;
262 
263  // enable all phase current loop controllers
264  for (_i=0; _i<BUCK_NO_OF_PHASES; _i++) {
265  buckInstance->i_loop[_i].controller->status.bits.enabled = true;
266  }
267  }
268  // IF control mode is set to unsupported control mode, return error
269  else
270  {
271  return(BUCK_OPSRET_ERROR);
272  }
273 
274  }
275 
276  // increment reference
277  buckInstance->startup.v_ramp.reference += buckInstance->startup.v_ramp.ref_inc_step;
278 
279  // check if ramp is complete
280  if (buckInstance->startup.v_ramp.reference > buckInstance->v_loop.reference)
281  {
282  // Set reference to the desired level
283  buckInstance->startup.v_ramp.reference = buckInstance->v_loop.reference;
284 
285  // Reconnect API reference to controller
286  buckInstance->v_loop.controller->Ports.ptrControlReference = &buckInstance->v_loop.reference;
287 
288  retval = BUCK_OPSRET_COMPLETE;
289 
290  }
291  else
292  // remain in this state until ramp is complete
293  {
294  retval = BUCK_OPSRET_REPEAT;
295  }
296 
297 
298  return(retval);
299 
300 }

315 volatile uint16_t SubState_IRampUp(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
316 {
317  volatile uint16_t retval=0;
318 
319  // Set BUSY bit until process is complete
320  buckInstance->status.bits.busy = true;
321 
322  // in average current mode if voltage limit is hit, increment
323  // voltage loop limit to ramp up current loop
324  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_ACMC)
325  {
326  // Increment maximum current limit
327  buckInstance->v_loop.controller->Limits.MaxOutput += buckInstance->startup.i_ramp.ref_inc_step;
328 
329  if (buckInstance->v_loop.controller->Limits.MaxOutput >= buckInstance->set_values.i_ref)
330  // check if ramp is complete
331  {
332  buckInstance->v_loop.maximum = buckInstance->set_values.i_ref;
333  buckInstance->v_loop.controller->Limits.MaxOutput = buckInstance->v_loop.maximum;
334  retval = BUCK_OPSRET_COMPLETE;
335  }
336  else
337  // if ramp is not complete yet, remain in this state
338  {
339  retval = BUCK_OPSRET_REPEAT;
340  }
341  }
342  else // Non-Current Loops Ending up here need to be lifted to PG_DELAY
343  {
344  buckInstance->v_loop.controller->Limits.MaxOutput = buckInstance->v_loop.maximum;
345  retval = BUCK_OPSRET_COMPLETE;
346  }
347 
348  return(retval);
349 
350 }

365 volatile uint16_t SubState_PowerGoodDelay(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
366 {
367  volatile uint16_t retval=0;
368 
369  // Set BUSY bit until process is complete
370  buckInstance->status.bits.busy = true;
371 
372  // increment delay counter until the GOWER GOOD delay has expired
373  if(buckInstance->startup.power_good_delay.counter++ > buckInstance->startup.power_good_delay.period)
374  {
375  buckInstance->startup.power_good_delay.counter =
376  (buckInstance->startup.power_good_delay.period + 1); // Clamp to PERIOD_EXPIRED for future startups
377 
378  // If defined, set POWER_GOOD output
379  if(buckInstance->gpio.PowerGood.enabled)
380  {
381  retval = buckGPIO_Set(&buckInstance->gpio.PowerGood);
382  if(!retval)
383  return(BUCK_OPSTATE_ERROR);
384  }
385 
386  retval = BUCK_OPSRET_COMPLETE;
387  }
388  else
389  // if period has not expired yet, remain in this state
390  {
391  retval = BUCK_OPSRET_REPEAT;
392  }
393 
394  return(retval);
395 
396 }
