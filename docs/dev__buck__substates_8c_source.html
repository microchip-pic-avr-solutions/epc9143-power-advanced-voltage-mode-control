62 volatile uint16_t SubState_PowerOnDelay(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
63 {
64  volatile uint16_t retval=0;
65 
66  // Set BUSY bit until process is complete
67  buckInstance->status.bits.busy = true;
68 
69  // delay startup until POWER ON DELAY has expired
70  if(buckInstance->startup.power_on_delay.counter++ > buckInstance->startup.power_on_delay.period)
71  {
72  // Clamp POD counter to EXPIRED
73  buckInstance->startup.power_on_delay.counter =
74  (buckInstance->startup.power_on_delay.period + 1); // Saturate power on counter
75 
76  retval = BUCK_OPSRET_COMPLETE;
77  }
78  else
79  // When the period has not expired yet, return to this function
80  {
81  retval = BUCK_OPSRET_REPEAT;
82  }
83 
84  return(retval);
85 }

108 volatile uint16_t SubState_PrepareVRampUp(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
109 {
110  volatile uint16_t _i=0;
111  volatile uint32_t _vout=0, _vin=0, _start_dc=0;
112 
113  // Set BUSY bit until process is complete
114  buckInstance->status.bits.busy = true;
115 
116  // Hijack voltage loop controller reference
117  buckInstance->startup.v_ramp.reference = 0; // Reset Soft-Start Voltage Reference
118  buckInstance->startup.i_ramp.reference = 0; // Reset Soft-Start Current Reference
119  buckInstance->v_loop.controller->Ports.ptrControlReference =
120  &buckInstance->startup.v_ramp.reference; // Voltage loop is pointing to Soft-Start Reference
121 
122  // Pre-charge reference
123  // Never start above the pre-biased output voltage.
124  // Always start at or slightly below the pre-biased output voltage
125  buckInstance->startup.v_ramp.reference = buckInstance->data.v_out;
126 
127  // In average current mode, set current reference limit to max startup current level
128  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_ACMC)
129  { // Disable all current control loops and reset control loop histories
130  buckInstance->v_loop.maximum = buckInstance->set_values.i_ref;
131  buckInstance->v_loop.controller->Limits.MaxOutput = buckInstance->v_loop.maximum;
132  }
133 
134  // Pre-charge PWM and control loop history
135  if(((buckInstance->data.v_in - buckInstance->feedback.ad_vin.scaling.offset) > 0) &&
136  ((buckInstance->data.v_out - buckInstance->feedback.ad_vout.scaling.offset) > 0) )
137  {
138  _vout = __builtin_muluu(
139  (buckInstance->data.v_out - buckInstance->feedback.ad_vout.scaling.offset),
140  buckInstance->feedback.ad_vout.scaling.factor);
141  _vout >>= (16 - buckInstance->feedback.ad_vout.scaling.scaler);
142 
143  _vin = __builtin_muluu(
144  (buckInstance->data.v_in - buckInstance->feedback.ad_vin.scaling.offset),
145  buckInstance->feedback.ad_vin.scaling.factor);
146  _vin >>= (16 - buckInstance->feedback.ad_vin.scaling.scaler);
147 
148  // Protect against negative duty cycle results
149  if (_vout > _vin) _vout = _vin;
150 
151  // CALCULATE BUCK CONVERTER STARTUP DUTY RATIO
152  // DC = VOUT / VIN, where DC = D * PERIOD
153 
154  if(_vin > 0)
155  {
156  _start_dc = __builtin_muluu((_vout), buckInstance->sw_node[0].period);
157  _start_dc = __builtin_divud(_start_dc, (uint16_t)_vin);
158  }
159  else
160  { _start_dc = (uint16_t)buckInstance->sw_node[_i].duty_ratio_min; }
161  }
162  else
163  // If there is no input voltage or no output voltage, start with minimum duty ratio
164  {
165  for (_i=0; _i<buckInstance->set_values.no_of_phases; _i++) {
166  _start_dc = (uint16_t)buckInstance->sw_node[_i].duty_ratio_min;
167  }
168  }
169 
170  // Program PWM module and pre-charge control loop
171  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_VMC)
172  {
173  if(_start_dc < buckInstance->v_loop.minimum)
174  { _start_dc = buckInstance->v_loop.minimum; }
175  else if(_start_dc > buckInstance->v_loop.maximum)
176  { _start_dc = buckInstance->v_loop.maximum; }
177 
178  buckInstance->v_loop.ctrl_Precharge(buckInstance->v_loop.controller, 0, _start_dc);
179  *buckInstance->v_loop.controller->Ports.Target.ptrAddress = _start_dc; // set initial PWM duty ratio
180  if(buckInstance->v_loop.controller->Ports.AltTarget.ptrAddress != NULL)
181  *buckInstance->v_loop.controller->Ports.AltTarget.ptrAddress = _start_dc; // set initial PWM duty ratio
182 
183  }
184  else if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_ACMC)
185  {
186  // Limit startup duty cycle to largest minimum/smallest maximum
187  for (_i=0; _i<buckInstance->set_values.no_of_phases; _i++) {
188 
189  if(_start_dc < buckInstance->i_loop[_i].minimum)
190  { _start_dc = buckInstance->i_loop[_i].minimum; }
191  else if(_start_dc > buckInstance->i_loop[_i].maximum)
192  { _start_dc = buckInstance->i_loop[_i].maximum; }
193  }
194 
195  // Set initial duty cycle
196  for (_i=0; _i<buckInstance->set_values.no_of_phases; _i++)
197  {
198  // pre-charge current loop control histories with ideal duty cycle
199  buckInstance->i_loop[_i].ctrl_Precharge(
200  buckInstance->i_loop[_i].controller, 0, _start_dc
201  );
202 
203  // pre-charge PWM outputs with ideal duty cycle
204  *buckInstance->i_loop[_i].controller->Ports.Target.ptrAddress = _start_dc; // set initial PWM duty ratio
205  if(buckInstance->i_loop[_i].controller->Ports.AltTarget.ptrAddress != NULL)
206  *buckInstance->i_loop[_i].controller->Ports.AltTarget.ptrAddress = _start_dc; // set initial PWM duty ratio
207  }
208  }
209  else
210  {
211  return(BUCK_OPSRET_ERROR);
212  }
213 
214  return(BUCK_OPSRET_COMPLETE);
215 
216 }

235 volatile uint16_t SubState_VRampUp(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
236 {
237  volatile uint16_t retval=1;
238  volatile uint16_t _i=0;
239 
240  // Set BUSY bit until process is complete
241  buckInstance->status.bits.busy = true;
242 
243  // ensure control loop is enabled
244  if(!buckInstance->v_loop.controller->status.bits.enabled) {
245 
246  // Enable all PWM channels
247  retval &= buckPWM_Resume(buckInstance); // Enable PWM outputs
248 
249  // In voltage mode control...
250  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_VMC)
251  { // only enable voltage loop
252  buckInstance->v_loop.controller->status.bits.enabled = true; // enable voltage loop controller
253  }
254  // In average current mode control...
255  else if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_ACMC)
256  {
257  // enable voltage loop controller
258  buckInstance->v_loop.controller->status.bits.enabled = true;
259 
260  // enable all phase current loop controllers
261  for (_i=0; _i<buckInstance->set_values.no_of_phases; _i++) {
262  buckInstance->i_loop[_i].controller->status.bits.enabled = true;
263  }
264  }
265  // IF control mode is set to unsupported control mode, return error
266  else
267  {
268  return(BUCK_OPSRET_ERROR);
269  }
270 
271  }
272 
273  // increment reference
274  buckInstance->startup.v_ramp.reference += buckInstance->startup.v_ramp.ref_inc_step;
275 
276  // check if ramp is complete
277  if (buckInstance->startup.v_ramp.reference > buckInstance->v_loop.reference)
278  {
279  // Set reference to the desired level
280  buckInstance->startup.v_ramp.reference = buckInstance->v_loop.reference;
281 
282  // Reconnect API reference to controller
283  buckInstance->v_loop.controller->Ports.ptrControlReference = &buckInstance->v_loop.reference;
284 
285  retval = BUCK_OPSRET_COMPLETE;
286 
287  }
288  else
289  // remain in this state until ramp is complete
290  {
291  retval = BUCK_OPSRET_REPEAT;
292  }
293 
294 
295  return(retval);
296 
297 }

314 volatile uint16_t SubState_IRampUp(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
315 {
316  volatile uint16_t retval=0;
317 
318  // Set BUSY bit until process is complete
319  buckInstance->status.bits.busy = true;
320 
321  // in average current mode if voltage limit is hit, increment
322  // voltage loop limit to ramp up current loop
323  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_ACMC)
324  {
325  // Increment maximum current limit
326  buckInstance->v_loop.controller->Limits.MaxOutput += buckInstance->startup.i_ramp.ref_inc_step;
327 
328  if (buckInstance->v_loop.controller->Limits.MaxOutput >= buckInstance->set_values.i_ref)
329  // check if ramp is complete
330  {
331  buckInstance->v_loop.maximum = buckInstance->set_values.i_ref;
332  buckInstance->v_loop.controller->Limits.MaxOutput = buckInstance->v_loop.maximum;
333  retval = BUCK_OPSRET_COMPLETE;
334  }
335  else
336  // if ramp is not complete yet, remain in this state
337  {
338  retval = BUCK_OPSRET_REPEAT;
339  }
340  }
341  else // Non-Current Loops Ending up here need to be lifted to PG_DELAY
342  {
343  buckInstance->v_loop.controller->Limits.MaxOutput = buckInstance->v_loop.maximum;
344  retval = BUCK_OPSRET_COMPLETE;
345  }
346 
347  return(retval);
348 
349 }

366 volatile uint16_t SubState_PowerGoodDelay(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
367 {
368  volatile uint16_t retval=0;
369 
370  // Set BUSY bit until process is complete
371  buckInstance->status.bits.busy = true;
372 
373  // increment delay counter until the GOWER GOOD delay has expired
374  if(buckInstance->startup.power_good_delay.counter++ > buckInstance->startup.power_good_delay.period)
375  {
376  buckInstance->startup.power_good_delay.counter =
377  (buckInstance->startup.power_good_delay.period + 1); // Clamp to PERIOD_EXPIRED for future startups
378 
379  // If defined, set POWER_GOOD output
380  if(buckInstance->gpio.PowerGood.enabled)
381  {
382  retval = buckGPIO_Set(&buckInstance->gpio.PowerGood);
383  if(!retval)
384  return(BUCK_OPSTATE_ERROR);
385  }
386 
387  retval = BUCK_OPSRET_COMPLETE;
388  }
389  else
390  // if period has not expired yet, remain in this state
391  {
392  retval = BUCK_OPSRET_REPEAT;
393  }
394 
395  return(retval);
396 
397 }
