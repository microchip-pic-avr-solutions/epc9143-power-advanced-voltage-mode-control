71 volatile uint16_t SubState_PowerOnDelay(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
72 {
73  volatile uint16_t retval=0;
74 
75  // Set BUSY bit until process is complete
76  buckInstance->status.bits.busy = true;
77 
78  // delay startup until POWER ON DELAY has expired
79  if(buckInstance->startup.power_on_delay.counter++ > buckInstance->startup.power_on_delay.period)
80  {
81  // Clamp POD counter to EXPIRED
82  buckInstance->startup.power_on_delay.counter =
83  (buckInstance->startup.power_on_delay.period + 1); // Saturate power on counter
84 
85  retval = BUCK_OPSRET_COMPLETE;
86  }
87  else
88  // When the period has not expired yet, return to this function
89  {
90  retval = BUCK_OPSRET_REPEAT;
91  }
92 
93  return(retval);
94 }

115 volatile uint16_t SubState_PrepareVRampUp(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
116 {
117  volatile uint16_t _i=0;
118  volatile uint32_t _vout=0, _vin=0, _start_dc=0;
119 
120  // Set BUSY bit until process is complete
121  buckInstance->status.bits.busy = true;
122 
123  // Hijack voltage loop controller reference
124  buckInstance->startup.v_ramp.reference = 0; // Reset Soft-Start Voltage Reference
125  buckInstance->startup.i_ramp.reference = 0; // Reset Soft-Start Current Reference
126  buckInstance->v_loop.controller->Ports.ptrControlReference =
127  &buckInstance->startup.v_ramp.reference; // Voltage loop is pointing to Soft-Start Reference
128 
129  // Pre-charge reference
130  // Never start above the pre-biased output voltage.
131  // Always start at or slightly below the pre-biased output voltage
132  buckInstance->startup.v_ramp.reference = buckInstance->data.v_out;
133 
134  // In average current mode, set current reference limit to max startup current level
135  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_ACMC)
136  { // Disable all current control loops and reset control loop histories
137  buckInstance->v_loop.maximum = buckInstance->set_values.i_ref;
138  buckInstance->v_loop.controller->Limits.MaxOutput = buckInstance->v_loop.maximum;
139  }
140 
141  // Pre-charge PWM and control loop history
142  if(((buckInstance->data.v_in - buckInstance->feedback.ad_vin.scaling.offset) > 0) &&
143  ((buckInstance->data.v_out - buckInstance->feedback.ad_vout.scaling.offset) > 0) )
144  {
145  _vout = __builtin_muluu(
146  (buckInstance->data.v_out - buckInstance->feedback.ad_vout.scaling.offset),
147  buckInstance->feedback.ad_vout.scaling.factor);
148  _vout >>= (16 - buckInstance->feedback.ad_vout.scaling.scaler);
149 
150  _vin = __builtin_muluu(
151  (buckInstance->data.v_in - buckInstance->feedback.ad_vin.scaling.offset),
152  buckInstance->feedback.ad_vin.scaling.factor);
153  _vin >>= (16 - buckInstance->feedback.ad_vin.scaling.scaler);
154 
155  // Protect against negative duty cycle results
156  if (_vout < _vin) _vout = _vin;
157 
158  // CALCULATE BUCK CONVERTER STARTUP DUTY RATIO
159  // DC = (VOUT-VIN) / VOUT, where DC = D * PERIOD
160 
161  _start_dc = __builtin_muluu((_vout-_vin), buckInstance->sw_node[0].period);
162  _start_dc = __builtin_divud(_start_dc, (uint16_t)_vout);
163  }
164  else
165  // If there is no input voltage or no output voltage, start with minimum duty ratio
166  {
167  for (_i=0; _i<BUCK_NO_OF_PHASES; _i++) {
168  _start_dc = (uint16_t)buckInstance->sw_node[_i].duty_ratio_min;
169  }
170  }
171 
172  // Program PWM module and pre-charge control loop
173  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_VMC)
174  {
175  if(_start_dc < buckInstance->v_loop.minimum)
176  { _start_dc = buckInstance->v_loop.minimum; }
177  else if(_start_dc > buckInstance->v_loop.maximum)
178  { _start_dc = buckInstance->v_loop.maximum; }
179 
180  buckInstance->v_loop.ctrl_Precharge(buckInstance->v_loop.controller, 0, _start_dc);
181  *buckInstance->v_loop.controller->Ports.Target.ptrAddress = _start_dc; // set initial PWM duty ratio
182  if(buckInstance->v_loop.controller->Ports.AltTarget.ptrAddress != NULL)
183  *buckInstance->v_loop.controller->Ports.AltTarget.ptrAddress = _start_dc; // set initial PWM duty ratio
184 
185  }
186  else if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_ACMC)
187  {
188  // Limit startup duty cycle to largest minimum/smallest maximum
189  for (_i=0; _i<BUCK_NO_OF_PHASES; _i++) {
190 
191  if(_start_dc < buckInstance->i_loop[_i].minimum)
192  { _start_dc = buckInstance->i_loop[_i].minimum; }
193  else if(_start_dc > buckInstance->i_loop[_i].maximum)
194  { _start_dc = buckInstance->i_loop[_i].maximum; }
195  }
196 
197  // Set initial duty cycle
198  for (_i=0; _i<BUCK_NO_OF_PHASES; _i++)
199  {
200  // pre-charge current loop control histories with ideal duty cycle
201  buckInstance->i_loop[_i].ctrl_Precharge(
202  buckInstance->i_loop[_i].controller, 0, _start_dc
203  );
204 
205  // pre-charge PWM outputs with ideal duty cycle
206  *buckInstance->i_loop[_i].controller->Ports.Target.ptrAddress = _start_dc; // set initial PWM duty ratio
207  if(buckInstance->i_loop[_i].controller->Ports.AltTarget.ptrAddress != NULL)
208  *buckInstance->i_loop[_i].controller->Ports.AltTarget.ptrAddress = _start_dc; // set initial PWM duty ratio
209  }
210  }
211  else
212  {
213  return(BUCK_OPSRET_ERROR);
214  }
215 
216  return(BUCK_OPSRET_COMPLETE);
217 
218 }

236 volatile uint16_t SubState_VRampUp(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
237 {
238  volatile uint16_t retval=1;
239  volatile uint16_t _i=0;
240 
241  // Set BUSY bit until process is complete
242  buckInstance->status.bits.busy = true;
243 
244  // ensure control loop is enabled
245  if(!buckInstance->v_loop.controller->status.bits.enabled) {
246 
247  // Enable all PWM channels
248  retval &= buckPWM_Resume(buckInstance); // Enable PWM outputs
249 
250  // In voltage mode control...
251  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_VMC)
252  { // only enable voltage loop
253  buckInstance->v_loop.controller->status.bits.enabled = true; // enable voltage loop controller
254  }
255  // In average current mode control...
256  else if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_ACMC)
257  {
258  // enable voltage loop controller
259  buckInstance->v_loop.controller->status.bits.enabled = true;
260 
261  // enable all phase current loop controllers
262  for (_i=0; _i<BUCK_NO_OF_PHASES; _i++) {
263  buckInstance->i_loop[_i].controller->status.bits.enabled = true;
264  }
265  }
266  // IF control mode is set to unsupported control mode, return error
267  else
268  {
269  return(BUCK_OPSRET_ERROR);
270  }
271 
272  }
273 
274  // increment reference
275  buckInstance->startup.v_ramp.reference += buckInstance->startup.v_ramp.ref_inc_step;
276 
277  // check if ramp is complete
278  if (buckInstance->startup.v_ramp.reference > buckInstance->v_loop.reference)
279  {
280  // Set reference to the desired level
281  buckInstance->startup.v_ramp.reference = buckInstance->v_loop.reference;
282 
283  // Reconnect API reference to controller
284  buckInstance->v_loop.controller->Ports.ptrControlReference = &buckInstance->v_loop.reference;
285 
286  retval = BUCK_OPSRET_COMPLETE;
287 
288  }
289  else
290  // remain in this state until ramp is complete
291  {
292  retval = BUCK_OPSRET_REPEAT;
293  }
294 
295 
296  return(retval);
297 
298 }

313 volatile uint16_t SubState_IRampUp(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
314 {
315  volatile uint16_t retval=0;
316 
317  // Set BUSY bit until process is complete
318  buckInstance->status.bits.busy = true;
319 
320  // in average current mode if voltage limit is hit, increment
321  // voltage loop limit to ramp up current loop
322  if (buckInstance->set_values.control_mode == BUCK_CONTROL_MODE_ACMC)
323  {
324  // Increment maximum current limit
325  buckInstance->v_loop.controller->Limits.MaxOutput += buckInstance->startup.i_ramp.ref_inc_step;
326 
327  if (buckInstance->v_loop.controller->Limits.MaxOutput >= buckInstance->set_values.i_ref)
328  // check if ramp is complete
329  {
330  buckInstance->v_loop.maximum = buckInstance->set_values.i_ref;
331  buckInstance->v_loop.controller->Limits.MaxOutput = buckInstance->v_loop.maximum;
332  retval = BUCK_OPSRET_COMPLETE;
333  }
334  else
335  // if ramp is not complete yet, remain in this state
336  {
337  retval = BUCK_OPSRET_REPEAT;
338  }
339  }
340  else // Non-Current Loops Ending up here need to be lifted to PG_DELAY
341  {
342  buckInstance->v_loop.controller->Limits.MaxOutput = buckInstance->v_loop.maximum;
343  retval = BUCK_OPSRET_COMPLETE;
344  }
345 
346  return(retval);
347 
348 }

363 volatile uint16_t SubState_PowerGoodDelay(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
364 {
365  volatile uint16_t retval=0;
366 
367  // Set BUSY bit until process is complete
368  buckInstance->status.bits.busy = true;
369 
370  // increment delay counter until the GOWER GOOD delay has expired
371  if(buckInstance->startup.power_good_delay.counter++ > buckInstance->startup.power_good_delay.period)
372  {
373  buckInstance->startup.power_good_delay.counter =
374  (buckInstance->startup.power_good_delay.period + 1); // Clamp to PERIOD_EXPIRED for future startups
375 
376  // If defined, set POWER_GOOD output
377  if(buckInstance->gpio.PowerGood.enabled)
378  {
379  retval = buckGPIO_Set(&buckInstance->gpio.PowerGood);
380  if(!retval)
381  return(BUCK_OPSTATE_ERROR);
382  }
383 
384  retval = BUCK_OPSRET_COMPLETE;
385  }
386  else
387  // if period has not expired yet, remain in this state
388  {
389  retval = BUCK_OPSRET_REPEAT;
390  }
391 
392  return(retval);
393 
394 }
