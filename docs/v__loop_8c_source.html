30  * For optimized data processing during DSP computations, these arrays must be

100 /*@@v_loop_Initialize()
101  * *********************************************************************************
102  * Summary: Initializes controller coefficient arrays and normalization
103  *
104  * Parameters:
105  * - struct NPNZ16b_s* controller
106  *
107  * Returns:
108  * - uint16_t: 0->failure
109  * 1->success
110  *
111  * Description:
112  * This function needs to be called from user code once to initialize coefficient
113  * arrays and number normalization settings of the v_loop controller
114  * object.
115  *
116  * PLEASE NOTE:
117  * This routine DOES NOT initialize the complete controller object.
118  * User-defined settings such as pointers to the control reference, source and
119  * target registers, output minima and maxima and further, design-dependent
120  * settings, need to be specified in user code.
121  * ********************************************************************************/
122 
123 volatile uint16_t v_loop_Initialize(volatile struct NPNZ16b_s* controller)
124 {
125  volatile uint16_t i=0;
126 
127  // Initialize controller data structure at runtime with pre-defined default values
128  controller->status.value = NPNZ16_CONTROL_STATUS_CLEAR; // clear all status flag bits (will turn off execution))
129 
130  controller->Filter.ptrACoefficients = &v_loop_coefficients.ACoefficients[0]; // initialize pointer to A-coefficients array
131  controller->Filter.ptrBCoefficients = &v_loop_coefficients.BCoefficients[0]; // initialize pointer to B-coefficients array
132  controller->Filter.ptrControlHistory = &v_loop_histories.ControlHistory[0]; // initialize pointer to control history array
133  controller->Filter.ptrErrorHistory = &v_loop_histories.ErrorHistory[0]; // initialize pointer to error history array
134  controller->Filter.normPostShiftA = v_loop_post_shift_A; // initialize A-coefficients/single bit-shift scaler
135  controller->Filter.normPostShiftB = v_loop_post_shift_B; // initialize B-coefficients/dual/post scale factor bit-shift scaler
136  controller->Filter.normPostScaler = v_loop_post_scaler; // initialize control output value normalization scaling factor
137  controller->Filter.normPreShift = v_loop_pre_scaler; // initialize A-coefficients/single bit-shift scaler
138 
139  controller->Filter.ACoefficientsArraySize = v_loop_ACoefficients_size; // initialize A-coefficients array size
140  controller->Filter.BCoefficientsArraySize = v_loop_BCoefficients_size; // initialize A-coefficients array size
141  controller->Filter.ControlHistoryArraySize = v_loop_ControlHistory_size; // initialize control history array size
142  controller->Filter.ErrorHistoryArraySize = v_loop_ErrorHistory_size; // initialize error history array size
143 
144  // Load default set of A-coefficients from user RAM into controller A-array located in X-Space
145  for(i=0; i<controller->Filter.ACoefficientsArraySize; i++)
146  {
147  v_loop_coefficients.ACoefficients[i] = v_loop_ACoefficients[i]; // Load coefficient A1 value into v_loop coefficient data space
148  v_loop_coefficients.ACoefficients[i] = v_loop_ACoefficients[i]; // Load coefficient A2 value into v_loop coefficient data space
149  v_loop_coefficients.ACoefficients[i] = v_loop_ACoefficients[i]; // Load coefficient A3 value into v_loop coefficient data space
150  v_loop_coefficients.ACoefficients[i] = v_loop_ACoefficients[i]; // Load coefficient A4 value into v_loop coefficient data space
151  }
152 
153  // Load default set of B-coefficients from user RAM into controller B-array located in X-Space
154  for(i=0; i<controller->Filter.BCoefficientsArraySize; i++)
155  {
156  v_loop_coefficients.BCoefficients[i] = v_loop_BCoefficients[i]; // Load coefficient B0 value into v_loop coefficient data space
157  v_loop_coefficients.BCoefficients[i] = v_loop_BCoefficients[i]; // Load coefficient B1 value into v_loop coefficient data space
158  v_loop_coefficients.BCoefficients[i] = v_loop_BCoefficients[i]; // Load coefficient B2 value into v_loop coefficient data space
159  v_loop_coefficients.BCoefficients[i] = v_loop_BCoefficients[i]; // Load coefficient B3 value into v_loop coefficient data space
160  v_loop_coefficients.BCoefficients[i] = v_loop_BCoefficients[i]; // Load coefficient B4 value into v_loop coefficient data space
161  }
162 
163  // Clear error and control histories of the 3P3Z controller
164  v_loop_Reset(&v_loop);
165 
166  // Load P-Term factor and scaler into data structure
167  controller->Filter.PTermFactor = v_loop_pterm_factor;
168  controller->Filter.PTermScaler = v_loop_pterm_scaler;
169 
170  // Load initial AGC factor and scaler into data structure
171  controller->GainControl.AgcFactor = v_loop_agc_factor_default;
172  controller->GainControl.AgcScaler = v_loop_agc_scaler_default;
173 
174  return(1);
175 }
