29  * ************************************************************************************************
30  * Summary:
31  * Generic Peripheral Pin Select (PPS) Driver Module
32  *
33  * Dependencies:
34  * p33c_pps.h
35  *
36  * Description:
37  * Some low-pin count devices have the capability to assign digital functions to a range of
38  * free configurable pins (RPx). This driver module offers functions to read from and write to
39  * the configuration registers, which will connect a digital peripheral bock with a selected
40  * pin.
41  *
42  * References:
43  * dsPIC33F Reference Manual, Section 10: I/O Ports, Chapter 10.3: Peripheral Multiplexing
44  *
45  * Author: M91406
46  * Version: 1.3
47  *
48  * History:

57 /*@@PPS_LockIO
58  * ************************************************************************************************
59  * Summary:
60  * Locks the Peripheral Pin Select Configuration registers against accidental changes
61  *
62  * Parameters:
63  * (none)
64  *
65  * Returns:
66  * 1: Success
67  * 0: Error
68  *
69  * Description:
70  * This inline-assembly routine locks the Port Multiplexing Configuration registers by keeping
71  * the required number of cycles during the Lock pocess. This function has to be called once
72  * before digital functions are mapped to a specific pin. Once called, a series of assignments
73  * can be executed.
74  *
75  * Note:
76  * If bit IOL1WAY in the Oscillator Configuration Bits (FOSC) is set, only one pin assignment
77  * operation will be allowed. all following calls of Unlock() or Lock() will be ignored.
78  *
79  * See Also:
80  * p33c_pps.h
81  *
82  * ***********************************************************************************************/
83 volatile uint16_t PPS_LockIO(void){
84 
85  __builtin_write_RPCON(0x0800); // lock PPS
86  return(RPCONbits.IOLOCK);
87 
88 }

90 /*@@PPS_UnlockIO
91  * ************************************************************************************************
92  * Summary:
93  * Unlocks the Peripheral Pin Select Configuration registers to enable changes
94  *
95  * Parameters:
96  * (none)
97  *
98  * Returns:
99  * 1: Success
100  * 0: Error
101  *
102  * Description:
103  * This inline-assembly routine unlocks the Port Multiplexing Configuration registers by keeping
104  * the required number of cycles during the unlock process. This function has to be called once
105  * after digital functions have been mapped to a specific pin, to prevent accidental changes.
106  *
107  * Note:
108  * If bit IOL1WAY in the Oscillator Configuration Bits (FOSC) is set, only one pin assignment
109  * operation will be allowed. all following calls of Unlock() or Lock() will be ignored.
110  *
111  * See Also:
112  * p33c_pps.h
113  *
114  * ***********************************************************************************************/
115 volatile uint16_t PPS_UnlockIO(void){
116 
117  __builtin_write_RPCON(0x0000); // unlock PPS
118  return(1 - RPCONbits.IOLOCK);
119 
120 }

122 /*@@PPS_RemapOutput
123  * ************************************************************************************************
124  * Summary:
125  * Assigns a digital function output to a pin
126  *
127  * Parameters:
128  * unsigned char pinno: Number of the RPx-pin, which should be assigned to the function
129  * unsigned char peripheral: Function, which should be assigned to the pin
130  *
131  * Returns:
132  * 1: Success
133  * 0: Error
134  *
135  * Description:
136  * Any supported digital function output (e.g. UART TxD) can be assigned to one of the RPx pins
137  * of the MCU/DSC. To assign a function output to a pin, call
138  *
139  * smpsPPS_RemapOutput([RP-NUMBER], [FUNCTION])
140  *
141  * Example:
142  * lres |= smpsPPS_RemapOutput(PIN_RP9 , PPSOUT_SYNCO1); // Assign RP9 to PWMSyncClkOutput
143  *
144  * See Also:
145  * p33c_pps.h
146  *
147  * ***********************************************************************************************/
148 volatile uint16_t PPS_RemapOutput(volatile uint8_t pinno, volatile uint8_t peripheral){
149 
150  volatile uint16_t retval = 0;
151  volatile uint8_t *regptr;
152  volatile uint8_t pin_offset=0;
153 
154  pin_offset = (pinno - RP_PINNO_MIN); // Determine pin-offset
155  regptr = (volatile uint8_t *)&RPOR0; // get register block base address
156  regptr += (volatile uint8_t)pin_offset; // add offset
157  *regptr = (volatile uint8_t)peripheral; // copy configuration into register location
158  retval = (bool)(*regptr == (volatile uint8_t)peripheral); // Check if contents have been written correctly
159 
160  return (retval);
161 
162 }

188  * smpsPPS_UnmapInput, smpsPPS_UnmapOutput
189  *
190  * ***********************************************************************************************/
191 volatile uint16_t PPS_RemapInput(volatile uint8_t pinno, volatile uint8_t *peripheral)
192 {
193 
194  // Map selected pin function
195  *peripheral = pinno;
196 
197  return 1;

201 /*@@PPS_UnmapOutput
202  * ************************************************************************************************
203  * Summary:
204  * Disconnects a pin from a digital function output
205  *
206  * Parameters:
207  * unsigned char pinno: Number of the RPx-pin
208  *
209  * Returns:
210  * 1: Success
211  * 0: Error
212  *
213  * Description:
214  * An existing assignment between any RPx pin and a supported digital function output will be
215  * dissolved.
216  *
217  * smpsPPS_UnmapOutput([RP-NUMBER])
218  *
219  * Example:
220  * result = smpsPPS_UnmapOutput(PIN_RP9); // Dissolve RP9 assignment
221  *
222  * See Also:
223  * p33c_pps.h
224  *
225  * ***********************************************************************************************/
226 volatile uint16_t PPS_UnmapOutput(volatile uint8_t pinno)
227 {
228  volatile uint16_t retval=0;
229 
230  // Unmap selected pin function
231  retval = PPS_RemapOutput(pinno, PPSPIN_NULL);
232  return retval;
233 
234 }
