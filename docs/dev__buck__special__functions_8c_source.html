179 volatile uint16_t CurrentSenseOffsetCalibration(volatile struct BUCK_POWER_CONTROLLER_s *buckInstance)
180 {
181  volatile uint16_t _i=0;
182  volatile bool _complete=true;
183  volatile enum BUCK_OPSTATE_RETURNS_e retval = BUCK_OPSRET_ERROR;
184 
185  // if current sense calibration is disabled, return COMPLETE and leave
186  if(!buckInstance->status.bits.cs_calib_enable)
187  {
188  return((uint16_t)BUCK_OPSRET_COMPLETE); // Set return value to COMPLETE and leave function
189  }
190 
191  // Protect against floored Current Calibration Procedure
192  if ((!buckInstance->status.bits.adc_active) ||
193  (!buckInstance->status.bits.pwm_active)
194  )
195  { return((uint16_t)BUCK_OPSRET_REPEAT); } // Return REPEAT
196 
197 
198  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
199  // Perform Calibration Step
200 
201  for(_i = 0; _i<buckInstance->set_values.no_of_phases; _i++)
202  {
203  // Reset Calibration Data Buffers if Calibration is rerun
204  if ((calib_cs[_i].start && calib_cs[_i].stop))
205  {
206  // Clear calibration data buffer
207  calib_cs[_i].buffer = 0; // Clear buffer
208  calib_cs[_i].counter = 0; // Clear counter
209  calib_cs[_i].start = false; // Clear START flag bit
210  calib_cs[_i].stop = false; // Clear STOP flag bit
211  }
212 
213  // Collect data samples
214  if (++calib_cs[_i].counter < CS_CALIB_STEPS)
215  {
216  calib_cs[_i].start = true; // Clear START flag bit
217  calib_cs[_i].stop = false; // Clear STOP flag bit
218  calib_cs[_i].buffer += buckInstance->data.i_sns[_i]; // Read ADC offset value
219  }
220  // Collection of data samples is complete
221  else
222  {
223  calib_cs[_i].buffer >>= 3; // Divide accumulated ADC samples (calculate average)
224  calib_cs[_i].stop = true; // Set STOP flag bit
225 
226  // Write calibration data to buck converter data structure
227  buckInstance->i_loop[_i].feedback_offset = calib_cs[_i].buffer;
228 
229  }
230 
231  _complete &= calib_cs[_i].stop; // track STOP flag bits
232  }
233 
234  // Return REPEAT until calibration is complete
235  if (_complete)
236  {
237  retval = (uint16_t)BUCK_OPSRET_COMPLETE; // Return COMPLETE
238  }
239  else
240  { retval = (uint16_t)BUCK_OPSRET_REPEAT; } // Return REPEAT
241 
242  return(retval);
243 
244 }
