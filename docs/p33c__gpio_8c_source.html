30 /* @@p33c_GpioInstance__GetHandle
31  * ********************************************************************************
32  * Summary:
33  * Gets pointer to GPIO Instance SFR set
34  *
35  * Parameters:
36  * uint16_t pgInstance: Index of the selected GPIO Instance (1=Port A, 2=Port B, etc.)
37  *
38  * Returns:
39  * struct P33C_GPIO_INSTANCE_s:
40  * GPIO instance object of the selected GPIO instance
41  *
42  * Description:
43  * This function returns the pointer to a GPIO instance register set in
44  * Special Function Register memory space. This pointer can be used to directly
45  * write to/read from the Special Function Registers of a given peripheral
46  * instance.
47  *
48  * ********************************************************************************/
49 
50 volatile struct P33C_GPIO_INSTANCE_s* p33c_GpioInstance_GetHandle(volatile uint16_t gpio_Instance)
51 {
52  volatile struct P33C_GPIO_INSTANCE_s* gpio;
53 
54  // Capture Handle: set pointer to memory address of desired GPIO instance
55  gpio = (volatile struct P33C_GPIO_INSTANCE_s*)
56  ((volatile uint8_t*) &ANSELA + (gpio_Instance * P33C_GPIO_SFR_OFFSET));
57 
58  return(gpio);
59 }

61 /* @@p33c_GpioInstance_Dispose
62  * ********************************************************************************
63  * Summary:
64  * Resets all GPIO Instance registers to their RESET default values
65  *
66  * Parameters:
67  * (none)
68  *
69  * Returns:
70  * 0 = failure, disposing GPIO instance was not successful
71  * 1 = success, disposing GPIO instance was successful
72  *
73  * Description:
74  * This function clears all GPIO Instance registers to their
75  * default values set when the device comes out of RESET.
76  *
77  * Default configuration:
78  * - all outputs are set to logic functions
79  * - all analog functions are disabled
80  * - all pull-up and pull-down resistors are disabled
81  * - all GPIOs are operating in push-pull mode (open drain disabled)
82  * - all GPIOs are configured as input with their signal level HIGH
83  *
84  * ********************************************************************************/
85 
86 volatile uint16_t p33c_GpioInstance_Dispose(volatile uint16_t gpioInstance)
87 {
88  volatile uint16_t retval=1;
89 
90  retval = p33c_GpioInstance_ConfigWrite(gpioInstance, gpioConfigClear);
91 
92  return(retval);
93 }

95 /* @@p33c_GpioInstance_ConfigRead
96  * ********************************************************************************
97  * Summary:
98  * Read the current configuration from the GPIO instance registers
99  *
100  * Parameters:
101  * (none)
102  *
103  * Returns:
104  * 0 = failure, reading GPIO instance was not successful (returns NULL)
105  * n = success, reading GPIO instance was successful (returns 16-bit wide pointer)
106  *
107  * Description:
108  * This function reads all registers with their current configuration into

112  *
113  * ********************************************************************************/
114

117  volatile struct P33C_GPIO_INSTANCE_s* gpio;
118 
119  // Set pointer to memory address of desired GPIO instance
120  gpio = (volatile struct P33C_GPIO_INSTANCE_s*)
121  ((volatile uint8_t*) &ANSELA + (gpioInstance * P33C_GPIO_SFR_OFFSET));
122 
123  return(*gpio);

125 }

127 /* @@p33c_GpioInstance_ConfigWrite
128  * ********************************************************************************
129  * Summary:
130  * Writes a user-defined configuration to the GPIO instance registers
131  *
132  * Parameters:
133  * (none)
134  *
135  * Returns:
136  * 0 = failure, writing GPIO instance was not successful
137  * 1 = success, writing GPIO instance was successful
138  *
139  * Description:
140  * This function writes a user-defined GPIO instance configuration of type
141  * P33C_GPIO_INSTANCE_s to the GPIO instance registers. The
142  * individual register configurations have to be set in user-code
143  * before calling this function. To simplify the configuration process
144  * of standard functions, this driver provides templates, which can be
145  * loaded and written directly.
146  *
147  * ********************************************************************************/
148 
149 volatile uint16_t p33c_GpioInstance_ConfigWrite(
150  volatile uint16_t gpioInstance,
151  volatile struct P33C_GPIO_INSTANCE_s gpioConfig
152 )
153 {
154  volatile uint16_t retval=1;
155  volatile struct P33C_GPIO_INSTANCE_s* gpio;
156 
157  // Set pointer to memory address of desired GPIO instance
158  gpio = (volatile struct P33C_GPIO_INSTANCE_s*)
159  ((volatile uint8_t*) &ANSELA + (gpioInstance * P33C_GPIO_SFR_OFFSET));
160  *gpio = gpioConfig;
161 
162  return(retval);
163 
164 }

172 /* @@gpioConfigClear
173  * ********************************************************************************
174  * Summary:
175  * Default RESET configuration of one GPIO instance SFRs
176  *
177  * Data type:
178  * struct P33C_GPIO_INSTANCE_s:
179  * GPIO Instance Special Function Register (SFR) set
180  *
181  * Description:
182  * Default configuration of the GPIO instance SFRs with all its registers
183  * being reset to their default state when the device comes out of RESET.
184  * Programmers can use this template to reset (dispose) a previously used
185  * GPIO instance when it's not used anymore or to secure a known startup
186  * condition before writing individual configurations to its SFRs.
187  *
188  * *******************************************************************************/
189 
190 volatile struct P33C_GPIO_INSTANCE_s gpioConfigClear = {
191 
192  .ANSELx.value = 0x0000, // Disable all analog functions
193  .CNCONx.value = 0x0000, // Reset all change notification configurations
194  .CNEN0x.value = 0x0000, // Disable all change notification functions
195  .CNEN1x.value = 0x0000, // Disable all change notification functions
196  .CNFx.value = 0x0000, // Clear all change notification interrupt flags
197  .CNPDx.value = 0x0000, // Disable all dull-down resistors
198  .CNPUx.value = 0x0000, // Disable all dull-up resistors
199  .CNSTATx.value = 0x0000, // Clear all change notification status notifications
200  .LATx.value = 0x0000, // Set all IOs of selected instance LOW
201  .ODCx.value = 0x0000, // Clear all open-drain configurations of instance
202  .PORTx.value = 0x0000, // CLear port registers of all IOs of instance
203  .TRISx.value = 0x1111, // Set all IOs of instance to INPUT
204 
205  };
