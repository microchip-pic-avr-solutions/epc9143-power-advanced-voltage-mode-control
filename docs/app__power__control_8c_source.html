168 volatile uint16_t appPowerSupply_Execute(void)
169 {
170  volatile uint16_t retval=1;
171  volatile uint16_t _i=0;
172  volatile uint16_t i_dummy=0;
173 
174  // Capture most recent samples
175  buck.data.v_in = BUCK_VIN_ADCBUF;
176  buck.data.i_sns[0] = BUCK_ISNS1_ADCBUF;
177  buck.data.i_sns[1] = BUCK_ISNS2_ADCBUF;
178 
179  // Accumulate phase currents
180  for (_i=0; _i<buck.set_values.no_of_phases; _i++)
181  { i_dummy += buck.data.i_sns[_i]; }
182  buck.data.i_out = i_dummy; // Set output current value
183 
184  // Check conditional parameters and fault flags
185  // ToDo: remove - flag has been replaced by fault handler function
186 // buck.status.bits.power_source_detected = (bool)
187 // ((BUCK_VIN_UVLO_TRIP < buck.data.v_in) && (buck.data.v_in<BUCK_VIN_OVLO_TRIP));
188 
189  // Execute buck converter state machine
190  retval &= drv_BuckConverter_Execute(&buck);
191 
192  // Execute slower, advanced control options
193  appPowerSupply_CurrentSenseCalibration();
194 // appPowerSupply_CurrentBalancing();
195 
196  // Buck regulation error is only active while controller is running
197  // and while being tied to a valid reference
198  if((buck.state_id.bits.opstate_id >= BUCK_OPSTATE_RAMPUP) &&
199  (buck.state_id.bits.substate_id >= BUCK_OPSTATE_V_RAMP_UP))
200  {
201  fltobj_BuckRegErr.ReferenceObject.ptrObject = buck.v_loop.controller->Ports.ptrControlReference;
202  #if (PLANT_MEASUREMENT == false)
203  fltobj_BuckRegErr.Status.bits.Enabled = buck.v_loop.controller->status.bits.enabled;
204  #endif
205  }
206  else
207  {
208  fltobj_BuckRegErr.Status.bits.Enabled = false;
209  }
210 
211  return(retval);
212 }
