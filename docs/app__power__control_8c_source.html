96 volatile uint16_t appPowerSupply_Execute(void)
97 {
98  volatile uint16_t retval=1;
99  volatile uint16_t _i=0;
100  volatile uint16_t i_dummy=0;
101 
102  // Capture most recent samples
103  buck.data.v_in = BUCK_VIN_ADCBUF;
104  buck.data.i_sns[0] = BUCK_ISNS1_ADCBUF;
105  buck.data.i_sns[1] = BUCK_ISNS2_ADCBUF;
106 
107  // Accumulate phase currents
108  for (_i=0; _i<buck.set_values.no_of_phases; _i++)
109  { i_dummy += buck.data.i_sns[_i]; }
110  buck.data.i_out = i_dummy; // Set output current value
111 
112  // Execute buck converter state machine
113  retval &= drv_BuckConverter_Execute(&buck);
114 
115  // Execute slower, advanced control options
116 // appPowerSupply_CurrentBalancing();
117 
118  // Buck regulation error is only active while controller is running
119  // and while being tied to a valid reference
120  if((buck.state_id.bits.opstate_id >= BUCK_OPSTATE_RAMPUP) &&
121  (buck.state_id.bits.substate_id >= BUCK_OPSTATE_V_RAMP_UP))
122  {
123  fltobj_BuckRegErr.ReferenceObject.ptrObject = buck.v_loop.controller->Ports.ptrControlReference;
124  #if (PLANT_MEASUREMENT == false)
125  fltobj_BuckRegErr.Status.bits.Enabled = buck.v_loop.controller->status.bits.enabled;
126  #endif
127  }
128  else
129  {
130  fltobj_BuckRegErr.Status.bits.Enabled = false;
131  }
132 
133  return(retval);
134 }

305 inline void appPowerSupply_CurrentBalancing(void)
306 {
307  static int16_t offset=0;
308 
309  // Current balancing is only executed in nominal running mode
310  if(buck.state_id.bits.opstate_id != BUCK_OPSTATE_ONLINE)
311  return;
312 
313  // if current in phase #1 is higher than phase current #2...
314  if(buck.data.i_sns[0] > buck.data.i_sns[1])
315  { // .. add 1 tick to phase #2 duty cycle
316  offset = ((++offset>0x07) ? 0x07 : offset);
317  }
318  else
319  { // .. sub 1 tick from phase #2 duty cycle
320  offset = ((--offset<0x00) ? 0x00 : offset);
321  }
322 
323  buck.v_loop.controller->Ports.AltTarget.Offset = (uint16_t)offset;
324 
325 }
