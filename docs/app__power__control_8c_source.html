128 volatile uint16_t appPowerSupply_Execute(void)
129 {
130  volatile uint16_t retval=1;
131  volatile uint16_t _i=0;
132  volatile uint16_t i_dummy=0;
133 
134  // Capture most recent samples
135  buck.data.v_in = BUCK_VIN_ADCBUF;
136  buck.data.i_sns[0] = BUCK_ISNS1_ADCBUF;
137  buck.data.i_sns[1] = BUCK_ISNS2_ADCBUF;
138 
139  // Accumulate phase currents
140  for (_i=0; _i<buck.set_values.no_of_phases; _i++)
141  { i_dummy += buck.data.i_sns[_i]; }
142  buck.data.i_out = i_dummy; // Set output current value
143 
144  // Execute buck converter state machine
145  retval &= drv_BuckConverter_Execute(&buck);
146 
147  // Execute slower, advanced control options
148  appPowerSupply_CurrentSenseCalibration();
149 // appPowerSupply_CurrentBalancing();
150 
151  // Buck regulation error is only active while controller is running
152  // and while being tied to a valid reference
153  if((buck.state_id.bits.opstate_id >= BUCK_OPSTATE_RAMPUP) &&
154  (buck.state_id.bits.substate_id >= BUCK_OPSTATE_V_RAMP_UP))
155  {
156  fltobj_BuckRegErr.ReferenceObject.ptrObject = buck.v_loop.controller->Ports.ptrControlReference;
157  #if (PLANT_MEASUREMENT == false)
158  fltobj_BuckRegErr.Status.bits.Enabled = buck.v_loop.controller->status.bits.enabled;
159  #endif
160  }
161  else
162  {
163  fltobj_BuckRegErr.Status.bits.Enabled = false;
164  }
165 
166  return(retval);
167 }
