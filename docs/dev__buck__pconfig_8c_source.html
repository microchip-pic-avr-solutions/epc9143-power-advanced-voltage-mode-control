47  PG2CONLbits.ON = 0; // PWM Generator #2 Enable: PWM Generator is not enabled
48  PG3CONLbits.ON = 0; // PWM Generator #3 Enable: PWM Generator is not enabled
49  PG4CONLbits.ON = 0; // PWM Generator #4 Enable: PWM Generator is not enabled
50 
51  #if defined (PG5CONL)
52  PG5CONLbits.ON = 0; // PWM Generator #5 Enable: PWM Generator is not enabled
53  #endif
54  #if defined (PG6CONL)
55  PG6CONLbits.ON = 0; // PWM Generator #6 Enable: PWM Generator is not enabled
56  #endif
57  #if defined (PG7CONL)
58  PG7CONLbits.ON = 0; // PWM Generator #7 Enable: PWM Generator is not enabled
59  #endif
60  #if defined (PG8CONL)
61  PG8CONLbits.ON = 0; // PWM Generator #8 Enable: PWM Generator is not enabled
62  #endif
63 
64  // Write default PWM Module configuration to PWM module base registers
65  pwm = p33c_PwmModule_GetHandle();
66  retval &= p33c_PwmModule_ConfigWrite(buckPwmModuleConfig);

85 volatile uint16_t buckPWM_ChannelInitialize(volatile struct BUCK_POWER_CONTROLLER_s* buckInstance)
86 {
87  volatile uint16_t retval=1;
88  volatile uint16_t _i=0;
89 
90  volatile struct P33C_GPIO_INSTANCE_s* gpio;
91  volatile struct P33C_PWM_GENERATOR_s* pg;
92  volatile uint16_t pwm_Instance;
93  volatile uint16_t gpio_Instance;
94 
95  for (_i=0; _i<buckInstance->set_values.no_of_phases; _i++) {
96 
97  // LOAD PERIPHERAL INSTANCES FROM BUCK CONVERTER OBJECT
98  pwm_Instance = buckInstance->sw_node[_i].pwm_instance;
99  gpio_Instance = buckInstance->sw_node[_i].gpio_instance;
100 
101  // CAPTURE MEMORY ADDRESS OF GIVEN PWM GENERATOR INSTANCE
102  gpio = p33c_GpioInstance_GetHandle(gpio_Instance);
103 
104  // WRITE GPIO CONFIGURATION OF PWM OUTPUT(S)
105  gpio->LATx.value &= ~(0x0001 << buckInstance->sw_node[_i].gpio_high); // Clear PWMxH output LOW
106  gpio->LATx.value &= ~(0x0001 << buckInstance->sw_node[_i].gpio_low); // Clear PWMxL output LOW
107  gpio->TRISx.value &= ~(0x0001 << buckInstance->sw_node[_i].gpio_high); // Clear PWMxH output to OUTPUT
108  gpio->TRISx.value &= ~(0x0001 << buckInstance->sw_node[_i].gpio_low); // Clear PWMxL output to OUTPUT
109  gpio->CNPDx.value |= (0x0001 << buckInstance->sw_node[_i].gpio_high); // Enable intern pull down register (PWM1H)
110  gpio->CNPDx.value |= (0x0001 << buckInstance->sw_node[_i].gpio_low); // Enable intern pull down register (PWM1L)
111 
112  // COPY CONFIGURATION FROM TEMPLATE TO PWM GENERATOR x CONTROL REGISTERS
113  pg = p33c_PwmGenerator_GetHandle(pwm_Instance);
114  retval &= p33c_PwmGenerator_ConfigWrite(buckInstance->sw_node[_i].pwm_instance, buckPwmGeneratorConfig);
115 
116  // LOAD PWM GENERATOR TIMING SETTINGS FROM BUCK CONVERTER OBJECT
117  pg->PGxCONH.bits.MPERSEL = (uint16_t)buckInstance->sw_node[_i].master_period_enable;
118  pg->PGxCONL.bits.HREN = (uint16_t)buckInstance->sw_node[_i].high_resolution_enable;
119 
120  pg->PGxDC.value = buckInstance->sw_node[_i].duty_ratio_min; // PGxDC: PWM GENERATOR x DUTY CYCLE REGISTER
121  pg->PGxPER.value = buckInstance->sw_node[_i].period; // PGxPER: PWM GENERATOR x PERIOD REGISTER
122  pg->PGxDTL.value = buckInstance->sw_node[_i].dead_time_falling; // PGxDTL: PWM GENERATOR x DEAD-TIME REGISTER LOW
123  pg->PGxDTH.value = buckInstance->sw_node[_i].dead_time_rising; // PGxDTH: PWM GENERATOR x DEAD-TIME REGISTER HIGH
124  pg->PGxLEBL.value = buckInstance->sw_node[_i].leb_period; // PWM GENERATOR x LEADING-EDGE BLANKING REGISTER LOW
125 
126  // ToDo: PWM Synchronization needs to be more universaL
127  // PWM synchronization only work within groups of 4 (PG1-PG4 or PG5-PG8)
128  // Multiphase boost converter auto PWM phase synchronization
129  if ((_i == 0) && (buckInstance->set_values.no_of_phases > 1))
130  { // First phase is always master phase
131  pg->PGxCONH.bits.MSTEN = 1; // Enable Master synchronization mode
132  pg->PGxCONH.bits.SOCS = 0b0000; // Master PWM always triggers itself
133  pg->PGxEVTL.bits.PGTRGSEL = 0b011; // PGxTRIGC is always used as PWM synchronization trigger output
134  pg->PGxTRIGC.value = buckInstance->sw_node[_i+1].phase; // Set phase shift between master phase and first synchronized phase
135  pg->PGxCONH.bits.UPDMOD = 0b001; // Immediate update
136  pg->PGxEVTL.bits.UPDTRG = 0b11; // A write of the PGxTRIGA register automatically sets the UPDATE bit
137  }
138  else if ((0 < _i) && (_i < (buckInstance->set_values.no_of_phases-1)))
139  { // Every synchronized phase is synchronized to previous phase while being master to following
140  pg->PGxCONH.bits.MSTEN = 1; // Enable Master synchronization mode
141  pg->PGxCONH.bits.SOCS = buckInstance->sw_node[_i-1].pwm_instance; // synchronized PWM is always triggered by previous generator while providing trigger for following
142  pg->PGxEVTL.bits.PGTRGSEL = 0b011; // PGxTRIGC is always used as PWM synchronization trigger output
143  pg->PGxTRIGC.value = buckInstance->sw_node[_i+1].phase; // Set phase shift between master phase and first synchronized phase
144  pg->PGxCONH.bits.UPDMOD = 0b011; // Sync immediate update
145  pg->PGxEVTL.bits.UPDTRG = 0; // User must set the UPDREQ bit (PGxSTAT[3]) manually
146  }
147  else if ((0 < _i) && (_i == (buckInstance->set_values.no_of_phases-1)))
148  { // Last phase does not provide any master trigger
149  pg->PGxCONH.bits.MSTEN = 0; // Disable Master synchronization mode
150  pg->PGxCONH.bits.SOCS = buckInstance->sw_node[_i-1].pwm_instance; // synchronized PWM is always triggered by previous generator while providing trigger for following
151  pg->PGxEVTL.bits.PGTRGSEL = 0b000; // EOC is the general trigger output
152  pg->PGxTRIGC.value = 0x0000; // Clear phase shift value
153  pg->PGxCONH.bits.UPDMOD = 0b011; // Sync immediate update
154  pg->PGxEVTL.bits.UPDTRG = 0; // User must set the UPDREQ bit (PGxSTAT[3]) manually
155  }
156  else if (buckInstance->set_values.no_of_phases == 1)
157  { // This is only a single phase system (no PWM dependencies)
158  pg->PGxCONH.bits.MSTEN = 0; // Disable Master synchronization mode
159  pg->PGxCONH.bits.SOCS = 0b0000; // Master PWM always triggers itself
160  pg->PGxEVTL.bits.PGTRGSEL = 0b011; // PGxTRIGC is always used as PWM synchronization trigger output
161  pg->PGxTRIGC.value = 0x0000; // Clear phase shift between master phase and first synchronized phase
162  pg->PGxCONH.bits.UPDMOD = 0b001; // Immediate update
163  pg->PGxEVTL.bits.UPDTRG = 0b11; // A write of the PGxTRIGA register automatically sets the UPDATE bit
164  }
165 
166  // Update PWM generator timing registers
167  pg->PGxSTAT.bits.UPDREQ = 1; // Manually set the Update Request bit
168  pg->PGxCONH.bits.TRGMOD = 1; // all PWM generators are in retriggerable mode
169 
170  }
171 
172  return(retval);
173 }

185 volatile uint16_t buckPWM_Start(volatile struct BUCK_POWER_CONTROLLER_s* buckInstance)
186 {
187  volatile uint16_t retval=1;
188  volatile uint16_t _i=0;
189  volatile uint16_t timeout=0;
190  volatile uint16_t pwm_Instance=0;
191  volatile struct P33C_PWM_GENERATOR_s* pg;
192 
193  // Enable all PWM channels of the recent buck converter configuration
194  for(_i=0; _i< buckInstance->set_values.no_of_phases; _i++) {
195 
196  // Capture PWM instance of the recent channel
197  pwm_Instance = buckInstance->sw_node[_i].pwm_instance;
198 
199  // CAPTURE MEMORY ADDRESS OF GIVEN PWM GENERATOR INSTANCE
200  pg = p33c_PwmGenerator_GetHandle(pwm_Instance);
201 
202  pg->PGxIOCONL.value |= P33C_PGxIOCONL_OVREN; // PWMxH/L Output Override Enable: PWM generator controls the PWMxH output pin
203  pg->PGxIOCONH.value &= ~(P33C_PGxIOCONH_PEN); // PWMxH/L Output Port Disable: PWM generator controls the PWMxH output pin
204 
205  pg->PGxCONL.bits.ON = 1; // PWM Generator Enable: PWM Generator is enabled
206  pg->PGxSTAT.bits.UPDREQ = 1; // Update all PWM timing registers
207 
208  if(pg->PGxCONL.value & P33C_PGxCONL_HRES_EN) // If high resolution is enabled
209  {
210  while((!PCLKCONbits.HRRDY) && (timeout++ < 5000)); // wait for high resolution to get ready
211  if ((timeout >= 5000) || (PCLKCONbits.HRERR)) // if there is an error ...
212  return(0); // ... exit returning ERROR
213  }
214 
215  pg->PGxIOCONH.value |= P33C_PGxIOCONH_PEN; // PWMxH/L Output Port Enable: PWM generator controls the PWMxH output pin
216 
217  retval &= (volatile uint16_t)(pg->PGxCONL.bits.ON);
218 
219  }
220 
221  return(retval);
222 }

234 volatile uint16_t buckPWM_Stop(volatile struct BUCK_POWER_CONTROLLER_s* buckInstance)
235 {
236  volatile uint16_t retval=1;
237  volatile uint16_t _i=0;
238  volatile uint16_t pwm_Instance=0;
239  volatile struct P33C_PWM_GENERATOR_s* pg;
240 
241  // Disable all PWM channels of the recent buck converter configuration
242  for(_i=0; _i< buckInstance->set_values.no_of_phases; _i++) {
243 
244  // Capture PWM instance of the recent channel
245  pwm_Instance = buckInstance->sw_node[_i].pwm_instance;
246 
247  // CAPTURE MEMORY ADDRESS OF GIVEN PWM GENERATOR INSTANCE
248  pg = p33c_PwmGenerator_GetHandle(pwm_Instance);
249 
250  pg->PGxIOCONL.value |= P33C_PGxIOCONL_OVREN; // PWMxH/L Output Override Enable
251  pg->PGxIOCONH.value &= ~(P33C_PGxIOCONH_PEN); // PWMxH/L Output Pint Control Disable
252  pg->PGxCONL.value &= ~(P33C_PGxCONL_PWM_ON); // PWM Generator Disable
253  pg->PGxDC.value = buckInstance->sw_node[_i].duty_ratio_min; // Reset Duty Cycle
254  pg->PGxSTAT.value |= P33C_PGxSTAT_UPDREQ; // Set the Update Request bit to update PWM timing
255 
256  retval &= (volatile uint16_t)((volatile bool)(pg->PGxCONL.bits.ON == 0));
257 
258  }
259 
260  // If defined, reset POWER_GOOD output
261  if(buckInstance->gpio.PowerGood.enabled)
262  retval &= buckGPIO_Clear(&buckInstance->gpio.PowerGood);
263 
264  return(retval);
265 }

447 volatile uint16_t buckADC_ChannelInitialize(volatile BUCK_ADC_INPUT_SETTINGS_t* adc_Instance)
448 {
449  volatile uint16_t retval=1;
450  volatile uint8_t* ptrADCRegister;
451  volatile uint8_t bit_offset;
452 
453  // Initialize ADC input registers
454  if (adc_Instance->enabled) {
455 
456  // Write level trigger setting
457  if (adc_Instance->adc_input < 16) {
458  ADLVLTRGL |= ((uint16_t)(adc_Instance->level_trigger) << adc_Instance->adc_input);
459  ADEIEL |= ((uint16_t)(adc_Instance->early_interrupt_enable) << adc_Instance->adc_input);
460  ADIEL |= ((uint16_t)(adc_Instance->interrupt_enable) << adc_Instance->adc_input);
461  }
462  else if (adc_Instance->adc_input < 32) {
463  ADLVLTRGH |= ((uint16_t)(adc_Instance->level_trigger) << (adc_Instance->adc_input - 16));
464  ADEIEH |= ((uint16_t)(adc_Instance->early_interrupt_enable) << (adc_Instance->adc_input - 16));
465  ADIEH |= ((uint16_t)(adc_Instance->interrupt_enable) << (adc_Instance->adc_input - 16));
466  }
467  else {
468  return(0); // ADC input number out of range
469  }
470 
471  // write input mode setting
472  if (adc_Instance->adc_input < 8)
473  bit_offset = (2 * adc_Instance->adc_input);
474  else if (adc_Instance->adc_input < 16)
475  bit_offset = (2 * (adc_Instance->adc_input-8));
476  else if (adc_Instance->adc_input < 24)
477  bit_offset = (2 * (adc_Instance->adc_input-16));
478  else if (adc_Instance->adc_input < 32)
479  bit_offset = (2 * (adc_Instance->adc_input-24));
480  else
481  return(0); // ADC input number out of range
482 
483  ptrADCRegister = (volatile uint8_t *)
484  ((volatile uint8_t *)&ADMOD0L + (volatile uint8_t)(adc_Instance->adc_input >> 8));
485 
486  *ptrADCRegister |= ((unsigned int)adc_Instance->signed_result << bit_offset);
487  *ptrADCRegister |= ((unsigned int)adc_Instance->differential_input << (bit_offset + 1));
488 
489  // Write ADC trigger source setting
490  ptrADCRegister = (volatile uint8_t *)
491  ((volatile uint8_t *)&ADTRIG0L + (volatile uint8_t)adc_Instance->adc_input);
492 
493  *ptrADCRegister = (volatile uint8_t)adc_Instance->trigger_source;
494 
495  // Register ADC core to be active
496  switch (adc_Instance->adc_core) {
497  case 0:
498  adcore_mask |= ADC_CORE0_MASK_INDEX;
499  if (adc_Instance->differential_input)
500  adcore_diff_mask |= ADC_CORE0_MASK_INDEX;
501  break;
502  case 1:
503  adcore_mask |= ADC_CORE1_MASK_INDEX;
504  if (adc_Instance->differential_input)
505  adcore_diff_mask |= ADC_CORE1_MASK_INDEX;
506  break;
507  case 2:
508  adcore_mask |= ADC_CORE2_MASK_INDEX;
509  if (adc_Instance->differential_input)
510  adcore_diff_mask |= ADC_CORE2_MASK_INDEX;
511  break;
512  case 3:
513  adcore_mask |= ADC_CORE3_MASK_INDEX;
514  if (adc_Instance->differential_input)
515  adcore_diff_mask |= ADC_CORE3_MASK_INDEX;
516  break;
517  default:
518  adcore_mask |= ADC_SHRCORE_MASK_INDEX;
519  if (adc_Instance->differential_input)
520  adcore_diff_mask |= ADC_SHRCORE_MASK_INDEX;
