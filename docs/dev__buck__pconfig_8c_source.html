112 volatile uint16_t buckPWM_ChannelInitialize(volatile struct BUCK_POWER_CONTROLLER_s* buckInstance)
113 {
114  volatile uint16_t retval=1;
115  volatile uint16_t _i=0;
116 
117  volatile struct P33C_GPIO_INSTANCE_s* gpio;
118  volatile struct P33C_PWM_GENERATOR_s* pg;
119  volatile uint16_t pwm_Instance;
120  volatile uint16_t gpio_Instance;
121 
122  for (_i=0; _i<buckInstance->set_values.no_of_phases; _i++) {
123 
124  // LOAD PERIPHERAL INSTANCES FROM BUCK CONVERTER OBJECT
125  pwm_Instance = buckInstance->sw_node[_i].pwm_instance;
126  gpio_Instance = buckInstance->sw_node[_i].gpio_instance;
127 
128  // CAPTURE MEMORY ADDRESS OF GIVEN PWM GENERATOR INSTANCE
129  gpio = p33c_GpioInstance_GetHandle(gpio_Instance);
130 
131  // WRITE GPIO CONFIGURATION OF PWM OUTPUT(S)
132  gpio->LATx.value &= ~(0x0001 << buckInstance->sw_node[_i].gpio_high); // Clear PWMxH output LOW
133  gpio->LATx.value &= ~(0x0001 << buckInstance->sw_node[_i].gpio_low); // Clear PWMxL output LOW
134  gpio->TRISx.value &= ~(0x0001 << buckInstance->sw_node[_i].gpio_high); // Clear PWMxH output to OUTPUT
135  gpio->TRISx.value &= ~(0x0001 << buckInstance->sw_node[_i].gpio_low); // Clear PWMxL output to OUTPUT
136  gpio->CNPDx.value |= (0x0001 << buckInstance->sw_node[_i].gpio_high); // Enable intern pull down register (PWM1H)
137  gpio->CNPDx.value |= (0x0001 << buckInstance->sw_node[_i].gpio_low); // Enable intern pull down register (PWM1L)
138 
139  // COPY CONFIGURATION FROM TEMPLATE TO PWM GENERATOR x CONTROL REGISTERS
140  pg = p33c_PwmGenerator_GetHandle(pwm_Instance);
141  retval &= p33c_PwmGenerator_ConfigWrite(buckInstance->sw_node[_i].pwm_instance, buckPwmGeneratorConfig);
142 
143  // LOAD PWM GENERATOR TIMING SETTINGS FROM BUCK CONVERTER OBJECT
144  pg->PGxCONH.bits.MPERSEL = (uint16_t)buckInstance->sw_node[_i].master_period_enable;
145  pg->PGxCONL.bits.HREN = (uint16_t)buckInstance->sw_node[_i].high_resolution_enable;
146 
147  pg->PGxDC.value = buckInstance->sw_node[_i].duty_ratio_min; // PGxDC: PWM GENERATOR x DUTY CYCLE REGISTER
148  pg->PGxPER.value = buckInstance->sw_node[_i].period; // PGxPER: PWM GENERATOR x PERIOD REGISTER
149  pg->PGxDTL.value = buckInstance->sw_node[_i].dead_time_falling; // PGxDTL: PWM GENERATOR x DEAD-TIME REGISTER LOW
150  pg->PGxDTH.value = buckInstance->sw_node[_i].dead_time_rising; // PGxDTH: PWM GENERATOR x DEAD-TIME REGISTER HIGH
151  pg->PGxLEBL.value = buckInstance->sw_node[_i].leb_period; // PWM GENERATOR x LEADING-EDGE BLANKING REGISTER LOW
152 
153  // ToDo: PWM Synchronization needs to be more universaL
154  // PWM synchronization only work within groups of 4 (PG1-PG4 or PG5-PG8)
155  // Multiphase boost converter auto PWM phase synchronization
156  if ((_i == 0) && (buckInstance->set_values.no_of_phases > 1))
157  { // First phase is always master phase
158  pg->PGxCONH.bits.MSTEN = 1; // Enable Master synchronization mode
159  pg->PGxCONH.bits.SOCS = 0b0000; // Master PWM always triggers itself
160  pg->PGxEVTL.bits.PGTRGSEL = 0b011; // PGxTRIGC is always used as PWM synchronization trigger output
161  pg->PGxTRIGC.value = buckInstance->sw_node[_i+1].phase; // Set phase shift between master phase and first synchronized phase
162  pg->PGxCONH.bits.UPDMOD = 0b001; // Immediate update
163  pg->PGxEVTL.bits.UPDTRG = 0b11; // A write of the PGxTRIGA register automatically sets the UPDATE bit
164  }
165  else if ((0 < _i) && (_i < (buckInstance->set_values.no_of_phases-1)))
166  { // Every synchronized phase is synchronized to previous phase while being master to following
167  pg->PGxCONH.bits.MSTEN = 1; // Enable Master synchronization mode
168  pg->PGxCONH.bits.SOCS = buckInstance->sw_node[_i-1].pwm_instance; // synchronized PWM is always triggered by previous generator while providing trigger for following
169  pg->PGxEVTL.bits.PGTRGSEL = 0b011; // PGxTRIGC is always used as PWM synchronization trigger output
170  pg->PGxTRIGC.value = buckInstance->sw_node[_i+1].phase; // Set phase shift between master phase and first synchronized phase
171  pg->PGxCONH.bits.UPDMOD = 0b011; // Sync immediate update
172  pg->PGxEVTL.bits.UPDTRG = 0; // User must set the UPDREQ bit (PGxSTAT[3]) manually
173  }
174  else if ((0 < _i) && (_i == (buckInstance->set_values.no_of_phases-1)))
175  { // Last phase does not provide any master trigger
176  pg->PGxCONH.bits.MSTEN = 0; // Disable Master synchronization mode
177  pg->PGxCONH.bits.SOCS = buckInstance->sw_node[_i-1].pwm_instance; // synchronized PWM is always triggered by previous generator while providing trigger for following
178  pg->PGxEVTL.bits.PGTRGSEL = 0b000; // EOC is the general trigger output
179  pg->PGxTRIGC.value = 0x0000; // Clear phase shift value
180  pg->PGxCONH.bits.UPDMOD = 0b011; // Sync immediate update
181  pg->PGxEVTL.bits.UPDTRG = 0; // User must set the UPDREQ bit (PGxSTAT[3]) manually
182  }
183  else if (buckInstance->set_values.no_of_phases == 1)
184  { // This is only a single phase system (no PWM dependencies)
185  pg->PGxCONH.bits.MSTEN = 0; // Disable Master synchronization mode
186  pg->PGxCONH.bits.SOCS = 0b0000; // Master PWM always triggers itself
187  pg->PGxEVTL.bits.PGTRGSEL = 0b011; // PGxTRIGC is always used as PWM synchronization trigger output
188  pg->PGxTRIGC.value = 0x0000; // Clear phase shift between master phase and first synchronized phase
189  pg->PGxCONH.bits.UPDMOD = 0b001; // Immediate update
190  pg->PGxEVTL.bits.UPDTRG = 0b11; // A write of the PGxTRIGA register automatically sets the UPDATE bit
191  }
192 
193  // Update PWM generator timing registers
194  pg->PGxSTAT.bits.UPDREQ = 1; // Manually set the Update Request bit
195  pg->PGxCONH.bits.TRGMOD = 1; // all PWM generators are in retriggerable mode
196 
197  }
198 
199  return(retval);
200 }

214 volatile uint16_t buckPWM_Start(volatile struct BUCK_POWER_CONTROLLER_s* buckInstance)
215 {
216  volatile uint16_t retval=1;
217  volatile uint16_t _i=0;
218  volatile uint16_t timeout=0;
219  volatile uint16_t pwm_Instance=0;
220  volatile struct P33C_PWM_GENERATOR_s* pg;
221 
222  // Enable all PWM channels of the recent buck converter configuration
223  for(_i=0; _i< buckInstance->set_values.no_of_phases; _i++) {
224 
225  // Capture PWM instance of the recent channel
226  pwm_Instance = buckInstance->sw_node[_i].pwm_instance;
227 
228  // CAPTURE MEMORY ADDRESS OF GIVEN PWM GENERATOR INSTANCE
229  pg = p33c_PwmGenerator_GetHandle(pwm_Instance);
230 
231  pg->PGxIOCONL.value |= P33C_PGxIOCONL_OVREN; // PWMxH/L Output Override Enable: PWM generator controls the PWMxH output pin
232  pg->PGxIOCONH.value &= ~(P33C_PGxIOCONH_PEN); // PWMxH/L Output Port Disable: PWM generator controls the PWMxH output pin
233 
234  pg->PGxCONL.bits.ON = 1; // PWM Generator Enable: PWM Generator is enabled
235  pg->PGxSTAT.bits.UPDREQ = 1; // Update all PWM timing registers
236 
237  if(pg->PGxCONL.value & P33C_PGxCONL_HRES_EN) // If high resolution is enabled
238  {
239  while((!PCLKCONbits.HRRDY) && (timeout++ < 5000)); // wait for high resolution to get ready
240  if ((timeout >= 5000) || (PCLKCONbits.HRERR)) // if there is an error ...
241  return(0); // ... exit returning ERROR
242  }
243 
244  pg->PGxIOCONH.value |= P33C_PGxIOCONH_PEN; // PWMxH/L Output Port Enable: PWM generator controls the PWMxH output pin
245 
246  retval &= (volatile uint16_t)(pg->PGxCONL.bits.ON);
247 
248  }
249 
250  return(retval);
251 }

499 volatile uint16_t buckADC_ChannelInitialize(volatile struct BUCK_ADC_INPUT_SETTINGS_s* adcInstance)
500 {
501  volatile uint16_t retval=1;
502  volatile uint8_t* ptrADCRegister;
503  volatile uint8_t bit_offset;
504 
505  // Initialize ADC input registers
506  if (adcInstance->enabled) {
507 
508  // Write level trigger setting
509  if (adcInstance->adc_input < 16) {
510  ADLVLTRGL |= ((uint16_t)(adcInstance->level_trigger) << adcInstance->adc_input);
511  ADEIEL |= ((uint16_t)(adcInstance->early_interrupt_enable) << adcInstance->adc_input);
512  ADIEL |= ((uint16_t)(adcInstance->interrupt_enable) << adcInstance->adc_input);
513  }
514  else if (adcInstance->adc_input < 32) {
515  ADLVLTRGH |= ((uint16_t)(adcInstance->level_trigger) << (adcInstance->adc_input - 16));
516  ADEIEH |= ((uint16_t)(adcInstance->early_interrupt_enable) << (adcInstance->adc_input - 16));
517  ADIEH |= ((uint16_t)(adcInstance->interrupt_enable) << (adcInstance->adc_input - 16));
518  }
519  else {
520  return(0); // ADC input number out of range
521  }
522 
523  // write input mode setting
524  if (adcInstance->adc_input < 8)
525  bit_offset = (2 * adcInstance->adc_input);
526  else if (adcInstance->adc_input < 16)
527  bit_offset = (2 * (adcInstance->adc_input-8));
528  else if (adcInstance->adc_input < 24)
529  bit_offset = (2 * (adcInstance->adc_input-16));
530  else if (adcInstance->adc_input < 32)
531  bit_offset = (2 * (adcInstance->adc_input-24));
532  else
533  return(0); // ADC input number out of range
534 
535  ptrADCRegister = (volatile uint8_t *)
536  ((volatile uint8_t *)&ADMOD0L + (volatile uint8_t)(adcInstance->adc_input >> 8));
537 
538  *ptrADCRegister |= ((unsigned int)adcInstance->signed_result << bit_offset);
539  *ptrADCRegister |= ((unsigned int)adcInstance->differential_input << (bit_offset + 1));
540 
541  // Write ADC trigger source setting
542  ptrADCRegister = (volatile uint8_t *)
543  ((volatile uint8_t *)&ADTRIG0L + (volatile uint8_t)adcInstance->adc_input);
544 
545  *ptrADCRegister = (volatile uint8_t)adcInstance->trigger_source;
546 
547  // Register ADC core to be active
548  switch (adcInstance->adc_core) {
549  case 0:
550  adcore_mask |= ADC_CORE0_MASK_INDEX;
551  if (adcInstance->differential_input)
552  adcore_diff_mask |= ADC_CORE0_MASK_INDEX;
553  break;
554  case 1:
555  adcore_mask |= ADC_CORE1_MASK_INDEX;
556  if (adcInstance->differential_input)
557  adcore_diff_mask |= ADC_CORE1_MASK_INDEX;
558  break;
559  case 2:
